71 static int audioCallback(const void* ibuf, void* obuf, unsigned long frames, const PaStreamCallbackTimeInfo* time, PaStreamCallbackFlags, void* drv)
72 {
73 return static_cast<portaudio*>(drv)->processAudio(time->currentTime,
74 (float**)ibuf,
75 static_cast<float**>(obuf),
76 frames);
77 }

79 virtual int processAudio(PaTime current_time, float** ibuf, float** obuf, unsigned long frames)
80 {
81 AVOIDDENORMALS;
82
83 // Cleanup hardware outputs that are not used by DSP
84 for (int i = fDsp->getNumOutputs(); i < fDevNumOutChans; i++) {
85 memset(obuf[i], 0, sizeof(FAUSTFLOAT) * fBufferSize);
86 }
87
88 // Process samples
89 fDsp->compute(current_time * 1000000., frames, ibuf, obuf);
90 return paContinue;
91 }

122 bool init(const char* /*name*/, int numInputs, int numOutputs)
123 {
124 // Note that Pa_Initialize handled multiple times calls and
125 // must be matched with a corresponding call to Pa_Terminate
126 if (pa_error(Pa_Initialize())) {
127 return false;
128 }
129
130 const PaDeviceInfo* idev = Pa_GetDeviceInfo(Pa_GetDefaultInputDevice());
131 const PaDeviceInfo* odev = Pa_GetDeviceInfo(Pa_GetDefaultOutputDevice());
132
133 printf("DEVICE = %p || %p\n", idev, odev);
134
135 //In case there is no audio device, the function fails
136
137 if (idev == NULL) {
138 fDevNumInChans = 0;
139 } else {
140 fDevNumInChans = idev->maxInputChannels;
141 fInputParameters.device = Pa_GetDefaultInputDevice();
142 fInputParameters.sampleFormat = paFloat32 | paNonInterleaved;
143 fInputParameters.channelCount = fDevNumInChans;
144 fInputParameters.hostApiSpecificStreamInfo = 0;
145 }
146
147 if (odev == NULL) {
148 fDevNumOutChans = 0;
149 } else{
150 fDevNumOutChans = odev->maxOutputChannels;
151 fOutputParameters.device = Pa_GetDefaultOutputDevice();
152 fOutputParameters.sampleFormat = paFloat32 | paNonInterleaved;;
153 fOutputParameters.channelCount = fDevNumOutChans;
154 fOutputParameters.hostApiSpecificStreamInfo = 0;
155 }
156
157 // A DSP that has only outputs or only inputs forces the presence of an output or input device
158 if (numInputs == 0 && numOutputs != 0 && fDevNumOutChans == 0) {
159 printf("Devices not adapted to DSP\n");
160 return false;
161 }
162
163 if (numInputs != 0 && numOutputs == 0 && fDevNumInChans == 0) {
164 printf("Devices not adapted to DSP\n");
165 return false;
166 }
167
168 // If no device exists : the function fails
169 PaError err;
170 if ((err = Pa_IsFormatSupported(((fDevNumInChans > 0) ? &fInputParameters : 0),
171 ((fDevNumOutChans > 0) ? &fOutputParameters : 0), fSampleRate)) != 0) {
172 printf("stream format is not supported err = %d\n", err);
173 return false;
174 }
175
176 if (pa_error(Pa_OpenStream(&fAudioStream, ((fDevNumInChans > 0) ? &fInputParameters : 0),
177 ((fDevNumOutChans > 0) ? &fOutputParameters : 0),
178 fSampleRate, fBufferSize, paNoFlag, audioCallback, this))) {
179 return false;
180 }
181
182 return true;
183 }
184
185 void setDsp(dsp* DSP)
186 {
187 fDsp = DSP;
188 if (fDsp->getNumInputs() > fDevNumInChans || fDsp->getNumOutputs() > fDevNumOutChans) {
189 printf("DSP has %d inputs and %d outputs, physical inputs = %d physical outputs = %d \n",
190 fDsp->getNumInputs(), fDsp->getNumOutputs(),
191 fDevNumInChans, fDevNumOutChans);
192 fDsp = new dsp_adapter(fDsp, fDevNumInChans, fDevNumOutChans, fBufferSize);
193 }
194
195 fDsp->init(fSampleRate);
196 }
