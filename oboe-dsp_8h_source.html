62 void setThreadAffinity()
63 {
64 pid_t current_thread_id = gettid();
65 cpu_set_t cpu_set;
66 CPU_ZERO(&cpu_set);
67
68 // If the callback cpu ids aren't specified then bind to the current cpu
69 if (fCpuIds.empty()) {
70 int current_cpu_id = sched_getcpu();
71 __android_log_print(ANDROID_LOG_ERROR, "Faust", "Binding to current CPU ID %d", current_cpu_id);
72 CPU_SET(current_cpu_id, &cpu_set);
73 } else {
74 __android_log_print(ANDROID_LOG_ERROR, "Faust", "Binding to %d CPU IDs", static_cast<int>(fCpuIds.size()));
75 for (size_t i = 0; i < fCpuIds.size(); i++) {
76 int cpu_id = fCpuIds.at(i);
77 __android_log_print(ANDROID_LOG_ERROR, "Faust", "CPU ID %d added to cores set", cpu_id);
78 CPU_SET(cpu_id, &cpu_set);
79 }
80 }
81
82 int res = sched_setaffinity(current_thread_id, sizeof(cpu_set_t), &cpu_set);
83 if (res == 0) {
84 __android_log_print(ANDROID_LOG_ERROR, "Faust", "Thread affinity set");
85 } else {
86 __android_log_print(ANDROID_LOG_ERROR, "Faust", "Error setting thread affinity error no: %d", res);
87 }
88
89 fIsThreadAffinitySet = true;
90 }

112 oboe::DataCallbackResult onAudioReady(oboe::AudioStream* audioStream, void* outbuffer, int32_t framesWrite)
113 {
114 fBufferSize = framesWrite;
115 int64_t t1 = getTimeUsec();
116 if (!fIsThreadAffinitySet) setThreadAffinity();
117
118 // Duplex case
119 if (fDSP->getNumInputs() > 0) {
120
121 // Read inputs
122 float* inbuffer = static_cast<float*>(alloca(fDSP->getNumInputs() * sizeof(float) * framesWrite));
123 oboe::ResultWithValue<int32_t> res = fInputStream->read(inbuffer, framesWrite, 0);
124
125 if (res) {
126 int32_t framesRead = res.value();
127 fInputs->interleavedRead(inbuffer, framesWrite, fDSP->getNumInputs());
128
129 // Call compute only when 'read frames' == 'frames to write'
130 if (framesRead == framesWrite) {
131 fDSP->compute(framesWrite, fInputs->buffers(), fOutputs->buffers());
132 fOutputs->interleavedWrite(static_cast<float*>(outbuffer), framesWrite, fDSP->getNumOutputs());
133 } else {
134 __android_log_print(ANDROID_LOG_ERROR, "Faust", "onAudioReady framesRead %d framesWrite %d", framesRead, framesWrite);
135 }
136
137 } else {
138 __android_log_print(ANDROID_LOG_ERROR, "Faust", "onAudioReady fInputStream->read fails");
139 }
140
141 } else {
142
143 // Output only
144 fDSP->compute(framesWrite, nullptr, fOutputs->buffers());
145 fOutputs->interleavedWrite(static_cast<float*>(outbuffer), framesWrite, fDSP->getNumOutputs());
146 }
147
148 runControlCallbacks();
149
150 int64_t t2 = getTimeUsec();
151 fCPUTable[(fCPUTableIndex++)&(CPU_TABLE_SIZE-1)] = t2 - t1;
152 return oboe::DataCallbackResult::Continue;
153 }

162 bool oboeInit()
163 {
164 oboe::AudioStreamBuilder builder;
165
166 // Possibly set SR
167 if (fSampleRate != -1) {
168 builder.setSampleRate(fSampleRate);
169 }
170
171 if (fDSP->getNumInputs() > 0) {
172 if (builder.setDirection(oboe::Direction::Input)
173 ->setFormat(oboe::AudioFormat::Float)
174 ->setChannelCount(fDSP->getNumInputs())
175 ->setPerformanceMode(oboe::PerformanceMode::LowLatency)
176 ->setSharingMode(oboe::SharingMode::Exclusive)
177 ->openManagedStream(fInputStream) != oboe::Result::OK)
178 return false;
179 // Allocate the stream adapter
180 fInputs = new AudioChannels(4096, fDSP->getNumInputs());
181 // Keep the input SR
182 fSampleRate = fInputStream->getSampleRate();
183 } else {
184 fInputs = nullptr;
185 }
186
187 if (fDSP->getNumOutputs() > 0) {
188 // Common setup
189 builder.setDirection(oboe::Direction::Output)
190 ->setCallback(this)
191 ->setFormat(oboe::AudioFormat::Float)
192 ->setChannelCount(fDSP->getNumOutputs())
193 ->setPerformanceMode(oboe::PerformanceMode::LowLatency)
194 ->setSharingMode(oboe::SharingMode::Exclusive);
195
196 // If fSampleRate was given at initialisation time, or has been set when opening the input
197 if (fSampleRate != -1) {
198 // Force output to use the same SR as input, and possibly setup resampler
199 if (builder.setSampleRate(fSampleRate)
200 ->setSampleRateConversionQuality(oboe::SampleRateConversionQuality::Fastest)
201 ->openManagedStream(fOutputStream) != oboe::Result::OK)
202 return false;
203 } else {
204 // Open with the native SR
205 if (builder.openManagedStream(fOutputStream) != oboe::Result::OK) return false;
206 }
207
208 // Allocate the stream adapter
209 fOutputs = new AudioChannels(4096, fDSP->getNumOutputs());
210 // Keep the output SR
211 fSampleRate = fOutputStream->getSampleRate();
212 } else {
213 fOutputs = nullptr;
214 }
215 return true;
216 }

295 // Returns the average proportion of available CPU being spent inside the audio callbacks (between 0 and 1.0).
296 float getCPULoad()
297 {
298 float sum = 0.f;
299 for (int i = 0; i < CPU_TABLE_SIZE; i++) {
300 sum += fCPUTable[i];
301 }
302 return (sum/float(CPU_TABLE_SIZE))/(1000000.f*float(fBufferSize)/float(fSampleRate));
303 }
