26#define __MemoryReader__

28#include "faust/gui/Soundfile.h"

30/*
31 A 'MemoryReader' object can be used to prepare a set of sound resources in memory, to be used by SoundUI::addSoundfile.
32
33 A Soundfile* object will have to be filled with a list of sound resources: the fLength, fOffset, fSampleRate and fBuffers fields
34 have to be completed with the appropriate values, and will be accessed in the DSP object while running.
35 *
36 */
37
38// To adapt for a real case use
39
40#define SOUND_CHAN 2
41#define SOUND_LENGTH 4096
42#define SOUND_SR 44100
43
44struct MemoryReader : public SoundfileReader {
45
46 MemoryReader()
47 {}
48 virtual ~MemoryReader()
49 {}
50
58 virtual bool checkFile(const std::string& path_name) override { return true; }
59
68 virtual void getParamsFile(const std::string& path_name, int& channels, int& length) override
69 {
70 channels = SOUND_CHAN;
71 length = SOUND_LENGTH;
72 }
73
83 virtual void readFile(Soundfile* soundfile, const std::string& path_name, int part, int& offset, int max_chan) override
84 {
85 soundfile->fLength[part] = SOUND_LENGTH;
86 soundfile->fSR[part] = SOUND_SR;
87 soundfile->fOffset[part] = offset;
88
89 // Audio frames have to be written for each chan
90 if (soundfile->fIsDouble) {
91 for (int sample = 0; sample < SOUND_LENGTH; sample++) {
92 for (int chan = 0; chan < SOUND_CHAN; chan++) {
93 static_cast<double**>(soundfile->fBuffers)[chan][offset + sample] = 0.f;
94 }
95 }
96 } else {
97 for (int sample = 0; sample < SOUND_LENGTH; sample++) {
98 for (int chan = 0; chan < SOUND_CHAN; chan++) {
99 static_cast<float**>(soundfile->fBuffers)[chan][offset + sample] = 0.f;
100 }
101 }
102 }
103
104 // Update offset
105 offset += SOUND_LENGTH;
106 }
107
108};
109
110#endif
