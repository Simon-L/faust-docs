160 virtual void compute(int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
161 {
162 if (fPlayButton == FAUSTFLOAT(1) && fMutex.try_lock()) {
163
164 int rcount = std::min<int>(count, int(fInfo.frames - fCurFrames));
165 playSlice(rcount, fCurFrames, 0, outputs);
166
167 if (rcount < count) {
168 if (fLoopButton == FAUSTFLOAT(1)) {
169 // Loop buffer
170 playSlice(count - rcount, 0, rcount, outputs);
171 fCurFrames = count - rcount;
172 } else {
173 // Otherwise clear end of buffer and stops
174 clearSlice(count - rcount, rcount, outputs);
175 fCurFrames = FAUSTFLOAT(0);
176 fPlayButton = 0;
177 }
178 } else {
179 fCurFrames += count;
180 }
181
182 fMutex.unlock();
183
184 } else {
185 // Clear output
186 clearSlice(count, 0, outputs);
187 }
188 }

211class sound_memory_player : public sound_base_player {
212
213 private:
214
215 FAUSTFLOAT** fBuffer; // Buffer of non-interleased frames
216
217 void playSlice(int count, int src, int dst, FAUSTFLOAT** outputs)
218 {
219 for (int i = 0; i < fInfo.channels; i++) {
220 memcpy(&(outputs[i])[dst], &(fBuffer[i])[src], sizeof(FAUSTFLOAT) * count);
221 }
222 }
223
224 void setFrame(int frames)
225 {
226 // If position change
227 if (fSetFrames != fLastSetFrames
228 && std::abs(fSetFrames - fLastSetFrames) > BUFFER_SIZE
229 && std::abs(fSetFrames - fCurFrames) > BUFFER_SIZE) {
230 fMutex.lock();
231 fLastSetFrames = fSetFrames;
232 fCurFrames = fSetFrames;
233 fMutex.unlock();
234 }
235 }
236
237 public:
238
239 sound_memory_player(const std::string& filename):sound_base_player(filename)
240 {
241 fBuffer = new FAUSTFLOAT*[fInfo.channels];
242 for (int chan = 0; chan < fInfo.channels; chan++) {
243 fBuffer[chan] = new FAUSTFLOAT[fInfo.frames];
244 }
245
246 FAUSTFLOAT buffer[BUFFER_SIZE * fInfo.channels];
247 sf_count_t nbf, index = 0;
248
249 do {
250 // Read buffer
251 nbf = fReaderFun(fFile, buffer, BUFFER_SIZE);
252 // Deinterleave it
253 for (int chan = 0; chan < fInfo.channels; chan++) {
254 for (int frame = 0; frame < nbf; frame++) {
255 fBuffer[chan][index + frame] = buffer[frame * fInfo.channels + chan];
256 }
257 }
258 // Move write index
259 index += nbf;
260 } while (nbf == BUFFER_SIZE);
261 }
262
263 virtual ~sound_memory_player()
264 {
265 for (int i = 0; i < fInfo.channels; i++) {
266 delete [] fBuffer[i];
267 }
268 delete [] fBuffer;
269 sf_close(fFile);
270 }

272 sound_memory_player* clone() { return new sound_memory_player(fFileName); }

280class sound_dtd_player : public sound_base_player {
281
282 private:
283
284 ringbuffer_t* fBuffer; // Ringbuffer written in setFrame and read in playSlice
285
286 void playSlice(int count, int src, int dst, FAUSTFLOAT** outputs)
287 {
288 size_t read_space_frames = convertToFrames(ringbuffer_read_space(fBuffer));
289
290 if (read_space_frames >= count) {
291
292 // Read from ringbuffer
293 FAUSTFLOAT buffer[count * fInfo.channels];
294 ringbuffer_read(fBuffer, (char*)buffer, convertFromFrames(count));
295
296 // Deinterleave and write to output
297 for (int chan = 0; chan < fInfo.channels; chan++) {
298 for (int frame = 0; frame < count; frame++) {
299 outputs[chan][dst + frame] = buffer[frame * fInfo.channels + chan];
300 }
301 }
302
303 } else {
304 std::cerr << "PlaySlice : missing " << (count - read_space_frames) << " frames\n";
305 }
306 }
307
308 size_t readAndWrite(FAUSTFLOAT* buffer, size_t size)
309 {
310 size_t nbf = fReaderFun(fFile, buffer, size);
311 ringbuffer_write(fBuffer, (char*)buffer, convertFromFrames(nbf));
312 return nbf;
313 }
314
315 void setFrame(int frames)
316 {
317 // If position change
318 if (fSetFrames != fLastSetFrames
319 && std::abs(fSetFrames - fLastSetFrames) > BUFFER_SIZE
320 && std::abs(fSetFrames - fCurFrames) > BUFFER_SIZE) {
321 fMutex.lock();
322 // Reset ringbuffer
323 ringbuffer_reset(fBuffer);
324 // Reset fCurFrames and seek in the file
325 fLastSetFrames = fSetFrames;
326 sf_seek(fFile, fSetFrames, SEEK_SET);
327 fCurFrames = fSetFrames;
328 fMutex.unlock();
329 }
330
331 size_t write_space_frames = convertToFrames(ringbuffer_write_space(fBuffer));
332
333 // If ringbuffer has to be filled
334 if (write_space_frames > HALF_RING_BUFFER_SIZE) {
335 FAUSTFLOAT buffer[HALF_RING_BUFFER_SIZE * fInfo.channels];
336 // Tries to read and write HALF_RING_BUFFER_SIZE frames
337 size_t nbf = readAndWrite(buffer, HALF_RING_BUFFER_SIZE);
338 // End of file is reached
339 if (nbf < HALF_RING_BUFFER_SIZE) {
340 // Read RING_BUFFER_SIZE/2 - nbf frame from the beginning of file
341 sf_seek(fFile, 0, SEEK_SET);
342 readAndWrite(buffer, HALF_RING_BUFFER_SIZE - nbf);
343 }
344 }
345 }
346
347 size_t convertToFrames(size_t bytes) { return bytes / (sizeof(FAUSTFLOAT) * fInfo.channels); }
348 size_t convertFromFrames(size_t frames) { return frames * sizeof(FAUSTFLOAT) * fInfo.channels; }
349
350 public:
351
352 sound_dtd_player(const std::string& filename):sound_base_player(filename)
353 {
354 // Create ringbuffer
355 fBuffer = ringbuffer_create(RING_BUFFER_SIZE * fInfo.channels * sizeof(FAUSTFLOAT));
356
357 // Read first buffer
358 setFrame(0);
359 }
360
361 virtual ~sound_dtd_player()
362 {
363 ringbuffer_free(fBuffer);
364 sf_close(fFile);
365 }
366
367 sound_dtd_player* clone() { return new sound_dtd_player(fFileName); }
