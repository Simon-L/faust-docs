66template <typename REAL>
67class time_bench_real {
68
69 typedef std::vector<uint64_t>::const_iterator vtype;
70
71 protected:
72
73 int fMeasure;
74 int fCount;
75 int fSkip;
76
77 // These values are used to determine the number of ticks in a second
78 uint64_t fFirstRDTSC;
79 uint64_t fLastRDTSC;
80
81 // These tables contains the last fCount in ticks
82 uint64_t* fStarts;
83 uint64_t* fStops;
84
85 struct timeval fTv1;
86 struct timeval fTv2;
87
91 uint64_t getTicks()
92 {
93 #if defined(__x86_64__)
94 union {
95 uint32_t i32[2];
96 uint64_t i64;
97 } count;
98 __asm__ __volatile__("rdtsc" : "=a" (count.i32[0]), "=d" (count.i32[1]));
99 return count.i64;
100 #else
101 timespec res;
102 clock_gettime(CLOCK_MONOTONIC_RAW, &res);
103 return static_cast<uint64_t>(res.tv_sec) * 1e9 + static_cast<uint64_t>(res.tv_nsec);
104 #endif
105 }
106
110 double rdtscpersec()
111 {
112 // If the environment variable CLOCKSPERSEC is defined, we use it instead of our own measurement
113 char* str = getenv("CLOCKSPERSEC");
114 if (str) {
115 int64_t cps = int64_t(atoll(str));
116 if (cps > 1000000000) {
117 return cps;
118 }
119 }
120 return double(fLastRDTSC - fFirstRDTSC)
121 / (((double(fTv2.tv_sec) * 1000000. + double(fTv2.tv_usec)) - (double(fTv1.tv_sec) * 1000000. + double(fTv1.tv_usec)))
122 / 1000000.);
123 }
124
128 double rdtsc2sec(uint64_t clk)
129 {
130 return double(clk) / rdtscpersec();
131 }
132
133 double rdtsc2sec(double clk)
134 {
135 return clk / rdtscpersec();
136 }
137
143 double megapersec(int frames, int chans, uint64_t clk)
144 {
145 return (double(frames) * double(chans) * double(sizeof(REAL))) / (1024. * 1024. * rdtsc2sec(clk));
146 }
147
151 uint64_t meanValue(vtype a, vtype b)
152 {
153 uint64_t r = 0;
154 unsigned int n = 0;
155 while (a != b) { r += *a++; n++; }
156 return (n > 0) ? r/n : 0;
157 }
158
162 double standardDeviation(vtype a, vtype b)
163 {
164 double mean = meanValue(a, b);
165 double sum_sd = 0;
166 unsigned int n = 0;
167 while (a != b) {
168 sum_sd += std::pow(mean - double(*a++), 2.0);
169 n++;
170 }
171 return (n > 0) ? ((std::sqrt(sum_sd/double(n)) * 100)/mean) : 0;
172 }
173
174 public:
175
176 time_bench_real(int count, int skip)
177 {
178 fSkip = skip;
179 fCount = count;
180 fMeasure = 0;
181 fFirstRDTSC = 0;
182 fLastRDTSC = 0;
183 fStarts = new uint64_t[fCount];
184 fStops = new uint64_t[fCount];
185 }
186
187 virtual ~time_bench_real()
188 {
189 delete [] fStarts;
190 delete [] fStops;
191 }
192
193 void startMeasure() { fStarts[fMeasure % fCount] = getTicks(); }
194
195 void stopMeasure() { fStops[fMeasure % fCount] = getTicks(); fMeasure++; }
196
197 void openMeasure()
198 {
199 struct timezone tz;
200 gettimeofday(&fTv1, &tz);
201 fFirstRDTSC = getTicks();
202 fMeasure = 0;
203 }
204
205 void closeMeasure()
206 {
207 struct timezone tz;
208 gettimeofday(&fTv2, &tz);
209 fLastRDTSC = getTicks();
210 }
211
212 double measureDurationUsec()
213 {
214 return ((double(fTv2.tv_sec) * 1000000. + double(fTv2.tv_usec)) - (double(fTv1.tv_sec) * 1000000. + double(fTv1.tv_usec)));
215 }
216
220 std::pair<double, double> getStats(int bsize, int ichans, int ochans)
221 {
222 assert(fMeasure > fCount);
223 std::vector<uint64_t> V(fCount);
224 for (int i = 0; i < fCount; i++) {
225 V[i] = fStops[i] - fStarts[i];
226 }
227 sort(V.begin(), V.end());
228
229 // Mean and standard deviation of 50 best values (gives relatively stable results)
230 uint64_t meavalx = meanValue(V.begin(), V.begin() + 50);
231 double sd = standardDeviation(V.begin(), V.begin() + 50);
232 return std::make_pair(megapersec(bsize, ichans + ochans, meavalx), sd);
233 }
234
238 void printStats(const char* applname, int bsize, int ichans, int ochans)
239 {
240 assert(fMeasure > fCount);
241 std::vector<uint64_t> V(fCount);
242 for (int i = 0; i < fCount; i++) {
243 V[i] = fStops[i] - fStarts[i];
244 }
245 sort(V.begin(), V.end());
246
247 // Mean of 10 best values (gives relatively stable results)
248 uint64_t meaval00 = meanValue(V.begin(), V.begin() + 5);
249 uint64_t meaval25 = meanValue(V.begin() + fCount / 4 - 2, V.begin() + fCount / 4 + 3);
250 uint64_t meaval50 = meanValue(V.begin() + fCount / 2 - 2, V.begin() + fCount / 2 + 3);
251 uint64_t meaval75 = meanValue(V.begin() + 3 * fCount / 4 - 2, V.begin() + 3 * fCount / 4 + 3);
252 uint64_t meaval100 = meanValue(V.end() - 5, V.end());
253
254 // Printing
255 fprintf(stdout, "%\t%f\t%f\t%f\t%f\t%f\n", applname,
256 megapersec(bsize, ichans+ochans, meaval00),
257 megapersec(bsize, ichans+ochans, meaval25),
258 megapersec(bsize, ichans+ochans, meaval50),
259 megapersec(bsize, ichans+ochans, meaval75),
260 megapersec(bsize, ichans+ochans, meaval100));
261 }
262
263 bool isRunning() { return (fMeasure <= (fCount + fSkip)); }
264
265 int getCount()
266 {
267 return fMeasure;
268 }

343template <typename REAL>
344class measure_dsp_real : public decorator_dsp {
345
346 protected:
347
348 REAL** fInputs;
349 REAL** fAllInputs;
350 REAL** fOutputs;
351 REAL** fAllOutputs;
352 time_bench_real<REAL>* fBench;
353 int fBufferSize;
354 int fInputIndex;
355 int fOutputIndex;
356 int fCount;
357 bool fControl;
358 RandomControlUI fRandomUI;
359
360 void init()
361 {
362 fDSP->init(BENCH_SAMPLE_RATE);
363 fDSP->buildUserInterface(&fRandomUI);
364
365 fInputIndex = 0;
366 fOutputIndex = 0;
367
368 fInputs = new REAL*[fDSP->getNumInputs()];
369 fAllInputs = new REAL*[fDSP->getNumInputs()];
370 for (int i = 0; i < fDSP->getNumInputs(); i++) {
371 fAllInputs[i] = new REAL[fBufferSize * NBV];
372 fInputs[i] = fAllInputs[i];
373 // Write noise in inputs (to avoid 'speedup' effect due to null values)
374 int R0_0 = 0;
375 for (int j = 0; j < fBufferSize * NBV; j++) {
376 int R0temp0 = (12345 + (1103515245 * R0_0));
377 fAllInputs[i][j] = REAL(4.656613e-10f * R0temp0);
378 R0_0 = R0temp0;
379 }
380 }
381
382 fOutputs = new REAL*[fDSP->getNumOutputs()];
383 fAllOutputs = new REAL*[fDSP->getNumOutputs()];
384 for (int i = 0; i < fDSP->getNumOutputs(); i++) {
385 fAllOutputs[i] = new REAL[fBufferSize * NBV];
386 fOutputs[i] = fAllOutputs[i];
387 // Write zero in outputs
388 memset(fAllOutputs[i], 0, sizeof(REAL) * fBufferSize * NBV);
389 }
390 }
391
392 bool setRealtimePriority()
393 {
394 struct passwd* pw;
395 struct sched_param param;
396 int policy;
397 uid_t uid = getuid();
398 pw = getpwnam("root");
399 setuid(pw->pw_uid);
400
401 int err = pthread_getschedparam(pthread_self(), &policy, &param);
402 if (err != 0) {
403 fprintf(stderr, "setRealtimePriority : pthread_getschedparam res = %d\n", err);
404 }
405 policy = SCHED_RR;
406 param.sched_priority = 80;
407 err = pthread_setschedparam(pthread_self(), policy, &param);
408 if (err != 0) {
409 fprintf(stderr, "setRealtimePriority : pthread_setschedparam res = %d\n", err);
410 }
411
412 setuid(uid);
413 return (err != -1);
414 }

460 measure_dsp_real(dsp* dsp,
461 int buffer_size,
462 double duration_in_sec,
463 bool trace = true,
464 bool control = false,
465 int ds = 0,
466 int us = 0,
467 int filter = 0)
468 :decorator_dsp(), fBufferSize(buffer_size), fControl(control)
469 {
470 std::string error;
471 fDSP = createSRAdapter<REAL>(dsp, error, ds, us, filter);
472 init();
473
474 // Creates a first time_bench_real object to estimate the proper 'count' number of measure to do later
475 fBench = new time_bench_real<REAL>(1000, 10);
476 measure();
477 double duration = fBench->measureDurationUsec();
478 if (trace) {
479 fprintf(stdout, "Duration %f\n", (duration / 1e6));
480 if (control) fprintf(stdout, "Random control is on\n");
481 }
482 fCount = int(1000 * (duration_in_sec * 1e6 / duration));
483 delete fBench;
484
485 // Then allocate final time_bench_real object with proper 'count' parameter
486 fBench = new time_bench_real<REAL>(fCount, 10);
487 }

489 virtual ~measure_dsp_real()
490 {
491 for (int i = 0; i < fDSP->getNumInputs(); i++) {
492 delete [] fAllInputs[i];
493 }
494 delete [] fInputs;
495 delete [] fAllInputs;
496
497 for (int i = 0; i < fDSP->getNumOutputs(); i++) {
498 delete [] fAllOutputs[i];
499 }
500 delete [] fOutputs;
501 delete [] fAllOutputs;
502
503 delete fBench;
504 // DSP is deallocated by the decorator_dsp class.
505 }
506
510 virtual void compute(int count, REAL** inputs, REAL** outputs)
511 {
512 AVOIDDENORMALS;
513 // Possibly update all controllers
514 if (fControl) fRandomUI.update();
515 // Only measure the 'compute' method
516 fBench->startMeasure();
517 fDSP->compute(count, reinterpret_cast<FAUSTFLOAT**>(inputs), reinterpret_cast<FAUSTFLOAT**>(outputs));
518 fBench->stopMeasure();
519 }
520
521 virtual void compute(double date_usec, int count, REAL** inputs, REAL** outputs)
522 {
523 compute(count, inputs, outputs);
524 }
525
529 void computeAll()
530 {
531 do {
532 for (int i = 0; i < fDSP->getNumInputs(); i++) {
533 REAL* allinputs = fAllInputs[i];
534 fInputs[i] = &allinputs[fInputIndex * fBufferSize];
535 fInputIndex = (1 + fInputIndex) % NBV;
536 }
537 for (int i = 0; i < fDSP->getNumOutputs(); i++) {
538 REAL* alloutputs = fAllOutputs[i];
539 fOutputs[i] = &alloutputs[fOutputIndex * fBufferSize];
540 fOutputIndex = (1 + fOutputIndex) % NBV;
541 }
542 compute(0, fBufferSize, fInputs, fOutputs);
543 } while (fBench->isRunning());
544 }
545
549 void openMeasure() { fBench->openMeasure(); }
550
554 void closeMeasure() { fBench->closeMeasure(); }
555
556 double measureDurationUsec()
557 {
558 return fBench->measureDurationUsec();
559 }
560
561 void measure()
562 {
563 setRealtimePriority();
564 openMeasure();
565 computeAll();
566 closeMeasure();
567 }
568
572 std::pair<double, double> getStats()
573 {
574 return fBench->getStats(fBufferSize, fDSP->getNumInputs(), fDSP->getNumOutputs());
575 }
576
580 void printStats(const char* applname)
581 {
582 fBench->printStats(applname, fBufferSize, fDSP->getNumInputs(), fDSP->getNumOutputs());
583 }
584
585 bool isRunning() { return fBench->isRunning(); }
586
587 float getCPULoad()
588 {
589 return (fBench->measureDurationUsec() / 1000.0 * BENCH_SAMPLE_RATE) / (double(fBench->getCount()) * double(fBufferSize) * 1000.0);
590 }
591
592 int getCount() { return fCount; }
