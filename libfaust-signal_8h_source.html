125// To be used with getSigInterval/setSigInterval
126// see: https://stackoverflow.com/questions/27442885/syntax-error-with-stdnumeric-limitsmax
127struct Interval {
128 double fLo = std::numeric_limits<double>::lowest(); //< minimal value
129 double fHi = (std::numeric_limits<double>::max)(); //< maximal value
130 int fLSB = -24; //< lsb in bits
131
132 // To be used to set a full interval
133 Interval(double lo, double hi, int lsb):fLo(lo), fHi(hi), fLSB(lsb)
134 {}
135
136 // To be used to only set the LSB, with fLo and fHi taking default values
137 Interval(int lsb):fLSB(lsb)
138 {}
139};
140
141inline static std::ostream& operator<<(std::ostream& dst, const Interval& it)
142{
143 dst << "Interval [" << it.fLo << ", " << it.fHi << ", " << it.fLSB << "]";
144 return dst;
145}
