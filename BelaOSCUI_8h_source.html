88 void oscMessageReceivedAux(oscpkt::Message* msg)
89 {
90 std::string msgAdress = msg->addressPattern();
91 int paramIndex = fAPIUI.getParamIndex(msgAdress.c_str());
92 float floatArg;
93 if (msg->match(msgAdress).popFloat(floatArg).isOkNoMoreArgs() && paramIndex != -1) {
94 fAPIUI.setParamValue(paramIndex, floatArg);
95 // "get" message with correct address
96 } else if (msg->match("/get").isOkNoMoreArgs()) {
97 for (int p = 0; p < fAPIUI.getParamsCount(); ++p) {
98 // show data to console
99 rt_printf("%s %f to %f\n", fAPIUI.getParamAddress(p), fAPIUI.getParamMin(p), fAPIUI.getParamMax(p));
100 // set data by OSC
101 fSender.newMessage(std::string(fAPIUI.getParamAddress(p))).add(fAPIUI.getParamMin(p)).add(fAPIUI.getParamMax(p)).send();
102 }
103 // "hello" message
104 } else if (msg->match("/hello").isOkNoMoreArgs()) {
105 // show datat to console.
106 rt_printf("adresses:%s, in:%i, out:%i\n", fIP.c_str(), fInputPort, fOutputPort);
107 // set data by OSC
108 std::string s = fAPIUI.getParamAddress(0);
109 s.erase(0, 1);
110 s = s.substr(0, s.find("/"));
111 s.insert(0, 1, '/');
112 fSender.newMessage(s).add(std::string(fIP)).add(fInputPort).add(fOutputPort).send();
113 }
114 }

116 static void oscMessageReceived(oscpkt::Message* msg, const char* addr, void* arg)
117 {
118 static_cast<BelaOSCUI*>(arg)->oscMessageReceivedAux(msg);
119 }
120
121 bool run() override
122 {
123 fReceiver.setup(fInputPort, oscMessageReceived, this);
124 fSender.setup(fOutputPort, fIP.c_str());
125 rt_printf("initconnect\n");
126 if (fOSCItems.size() == 0) {
127 rt_printf("%i widgets, OSC Adresses:\n", fAPIUI.getParamsCount());
128 for (int p = 0; p < fAPIUI.getParamsCount(); ++p) {
129 rt_printf("%s %f to %f\n", fAPIUI.getParamAddress(p), fAPIUI.getParamMin(p), fAPIUI.getParamMax(p));
130 fOSCItems.push_back(new oscItem(&fSender, this, fAPIUI.getParamAddress(p), fAPIUI.getParamZone(p)));
131 }
132 }
133 rt_printf("connected\n");
134 return true;
135 }
