74 std::tuple<double, double, double> bench(int run)
75 {
76 // First call with fCount = -1 will be used to estimate fCount by giving the wanted measure duration
77 if (fCount == -1) {
78 measure_dsp_real<REAL> mes(fDSP, fBufferSize, 5., fTrace, fControl, fDownSampling, fUpSampling, fFilter);
79 mes.measure();
80 // fCount is kept from the first duration measure
81 fCount = mes.getCount();
82 std::pair<double, double> res = mes.getStats();
83 return std::make_tuple(res.first, res.second, mes.getCPULoad());
84 } else {
85 measure_dsp_real<REAL> mes(fDSP, fBufferSize, fCount, fTrace, fControl, fDownSampling, fUpSampling, fFilter);
86 for (int i = 0; i < run; i++) {
87 mes.measure();
88 std::pair<double, double> res = mes.getStats();
89 if (fTrace) {
90 fprintf(stdout, "%f MBytes/sec, SD : %f%% (DSP CPU : %f%% at %d Hz)\n", res.first, res.second, (mes.getCPULoad() * 100), int(BENCH_SAMPLE_RATE));
91 }
92 FAUSTBENCH_LOG<double>(res.first);
93 }
94 std::pair<double, double> res = mes.getStats();
95 return std::make_tuple(res.first, res.second, mes.getCPULoad());
96 }

248 bool computeOne(const TOption& item, int run, std::tuple<double, double, double>& res)
249 {
250 int argc = 0;
251 const char* argv[64];
252 for (int i = 0; i < item.size(); i++) {
253 argv[argc++] = item[i].c_str();
254 }
255 argv[argc] = nullptr; // NULL terminated argv
256
257 if (fInput == "") {
258 fFactory = createDSPFactoryFromFile(fFilename.c_str(), argc, argv, fTarget, fError, fOptLevel);
259 } else {
260 fFactory = createDSPFactoryFromString("FaustDSP", fInput, argc, argv, fTarget, fError, fOptLevel);
261 }
262
263 if (!fFactory) {
264 fprintf(stderr, "Cannot create factory : %s\n", fError.c_str());
265 return false;
266 }
267
268 fDSP = fFactory->createDSPInstance();
269 if (!fDSP) {
270 fprintf(stderr, "Cannot create instance...\n");
271 return false;
272 }
273
274 if (fTrace) printItem(item);
275
276 res = bench(run);
277 // fDSP is deallocated by bench calling measure_dsp
278
279 deleteDSPFactory(fFactory);
280 fFactory = nullptr;
281 fDSP = nullptr;
282
283 return true;
284 }

286 std::tuple<double, double, double, TOption> findOptimizedParametersAux(const TOptionTable& options)
287 {
288 std::vector<std::tuple<int, double, double, double> > table_res;
289 std::tuple<double, double, double> res = {0., 0., 0.};
290
291 for (int i = 0; i < options.size(); i++) {
292 if (computeOne(addArgvItems(options[i], fArgc, fArgv), fRun, res)) {
293 table_res.push_back(std::make_tuple(i, std::get<0>(res), std::get<1>(res), std::get<2>(res)));
294 } else {
295 fprintf(stderr, "computeOne error...\n");
296 }
297 }
298
299 sort(table_res.begin(), table_res.end(), compareFun);
300 return std::make_tuple(std::get<1>(table_res[0]),
301 std::get<2>(table_res[0]),
302 std::get<3>(table_res[0]),
303 options[std::get<0>(table_res[0])]);
304 }
305
306 static bool compareFun(std::tuple<int, double, double, double> i, std::tuple<int, double, double, double> j)
307 {
308 return (std::get<1>(i) > std::get<1>(j));
309 }

396 std::tuple<double, double, double, TOption> findOptimizedParameters()
397 {
398 if (fTrace) fprintf(stdout, "Discover best parameters option\n");
399 std::tuple<double, double, double, TOption> best_scal = findOptimizedParametersAux(fScalOptionsTable);
400 std::tuple<double, double, double, TOption> best_vec = findOptimizedParametersAux(fVecOptionsTable);
401
402 TOption best_vec_option = std::get<3>(best_vec);
403 if (fTrace) fprintf(stdout, "Refined with -mcd\n");
404 best_vec_option.push_back("-mcd");
405 best_vec_option.push_back("0");
406 TOptionTable refined_vec_table;
407 refined_vec_table.push_back(best_vec_option);
408 for (int size = 2; size <= fBufferSize; size *= 2) {
409 best_vec_option = std::get<3>(best_vec);
410 best_vec_option.push_back("-mcd");
411 best_vec_option.push_back(std::to_string(size));
412 refined_vec_table.push_back(best_vec_option);
413 }
414
415 best_vec = findOptimizedParametersAux(refined_vec_table);
416
417 // Compare best scalar and vector modes
418 std::tuple<double, double, double, TOption> best_res;
419 if (std::get<0>(best_scal) > std::get<0>(best_vec)) {
420 best_res = best_scal;
421 } else {
422 best_res = best_vec;
423 }
424
425 // Current best
426 TOptionTable options_table;
427 {
428 TOption best = std::get<3>(best_res);
429 options_table.push_back(best);
430 }
431
432 if (std::get<3>(best_res)[0] == "-vec") {
433 if (fTrace) fprintf(stdout, "Check with -ct 0\n");
434 // Add -ct 0
435 {
436 TOption best = std::get<3>(best_res);
437 best.push_back("-ct");
438 best.push_back("0");
439 options_table.push_back(best);
440 }
441 if (fTrace) fprintf(stdout, "Check with -g or -dfs\n");
442 // Add -g
443 {
444 TOption best = std::get<3>(best_res);
445 best.push_back("-g");
446 options_table.push_back(best);
447 }
448 // Add -dfs
449 {
450 TOption best = std::get<3>(best_res);
451 best.push_back("-dfs");
452 options_table.push_back(best);
453 }
454 // Add -g and -dfs
455 {
456 TOption best = std::get<3>(best_res);
457 best.push_back("-g");
458 best.push_back("-dfs");
459 options_table.push_back(best);
460 }
461 return findOptimizedParametersAux(options_table);
462 } else {
463 if (fTrace) fprintf(stdout, "Check with -ct 0\n");
464 // Add -ct 0
465 {
466 TOption best = std::get<3>(best_res);
467 best.push_back("-ct");
468 best.push_back("0");
469 options_table.push_back(best);
470 }
471 return findOptimizedParametersAux(options_table);
472 }
473 }
