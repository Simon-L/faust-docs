194 void colourMap(double v, uint8_t *rgb) {
195 // RGB-packed colour map
196 double rgb1[51] = {0,0,0,0.1,0.1,0.08,0.204,0.1561,0.1154,0.3303,0.1489,0.132,0.4903,0.1289,0.1412,0.5584,0.15,0.2393,0.552,0.188,0.5,0.4351,0.2853,0.6354,0.3264,0.3979,0.7697,0.1107,0.5283,0.8896,0,0.604,0.9378,0.0606,0.7257,0.5554,0.2485,0.8033,0.137,0.6381,0.8057,0.0459,1,0.7994,0,1,0.9264,0.4412,1,1,1};
197 v *= (5 - v)/4; // warp the scale slightly, since our colour-map skews dark
198
199 double index = v*16;
200 int lowIndex = std::min(std::floor(index), 15.0);
201 double rH = (index - lowIndex), rL = 1 - rH;
202
203 double *rgbLow = rgb1 + 3*lowIndex;
204 for (int c = 0; c < 3; ++c) {
205 rgb[c] = std::round(255*std::sqrt(rgbLow[c]*rgbLow[c]*rL + rgbLow[c + 3]*rgbLow[c + 3]*rH));
206 }
207 return;
208 }

212 void renderBytes() {
213 double scaleX = outputWidth > 1 ? (width - 1.0)/(outputWidth - 1.0) : (width - 1.0);
214 double scaleY = outputHeight > 1 ? (height - 1.0)/(outputHeight - 1.0) : (height - 1.0);
215 auto getScaledPixel = [&](int outX, int outY) {
216 double scaledSum = 0, counter = 0;
217 double inX = outX*scaleX;
218 double inY = outY*scaleY;
219 // Bidirectional interpolation, scaling up or down
220 double spanX = std::max(1.0, scaleX), spanY = std::max(1.0, scaleY);
221 for (int x = std::max<int>(0, std::ceil(inX - spanX)); x < std::min<int>(width, std::floor(inX + spanX)); ++x) {
222 double wx = 1 - std::abs(x - inX)/spanX;
223 wx *= wx*(3 - 2*wx);
224 for (int y = std::max<int>(0, std::ceil(inY - spanY)); y < std::min<int>(height, std::floor(inY + spanY)); ++y) {
225 double wy = 1 - std::abs(y - inY)/spanY;
226 wy *= wy*(3 - 2*wy);
227 double w = wx*wy;
228 double v = std::max(0.0, std::min(1.0, scale.map((*this)(x, y))));
229 scaledSum += v*w;
230 counter += w;
231 }
232 }
233 return scaledSum/counter;
234 };

263 rowBytes[x + 1] = (byte - predicted);

323 void writeCode(uint32_t v, int bits, bool flipped=true) {
324 // DEFLATE codes are written MSB first, so we flip all the codes around
325 for (int b = 0; b < bits; ++b) {
326 int b2 = flipped ? (bits - 1 - b) : b;
327 pending |= ((v>>b2)&1)<<pendingBits;
328 ++pendingBits;
329 }
330 while (pendingBits >= 8) {
331 pngBytes.push_back(pending&0xff);
332 pending = pending>>8;
333 pendingBits -= 8;
334 }

336 void deflate(const unsigned char *block, int length, bool isFinal) {
337 for (int i = 0; i < length; ++i) {
338 uint32_t c = block[i];
339 adlerA = (adlerA + c)%65521;
340 adlerB = (adlerB + adlerA)%65521;
341 }
342
343 writeCode(2 + isFinal, 3, false); // fixed-code Huffman, with final-block flag
344 // Very limited search for repeating sections
345 int writeIndex = 0;
346 while (writeIndex < length) {
347 int bestDistance = 0, bestLength = 1;
348 for (int d = 1; d <= 4 && d < writeIndex; ++d) {
349 int sequenceLength = 0;
350 while (writeIndex + sequenceLength < length && sequenceLength < 18) {
351 int sourceIndex = writeIndex - d + (sequenceLength%d);
352 if (block[writeIndex + sequenceLength] != block[sourceIndex]) break;
353 ++sequenceLength;
354 }
355 if (sequenceLength > bestLength) {
356 bestLength = sequenceLength;
357 bestDistance = d;
358 }
359 }
360 if (bestLength >= 3) {
361 if (bestLength <= 10) {
362 writeCode(bestLength - 2, 7);
363 } else { // we limit the sequences to 18
364 int extra = (bestLength - 11);
365 writeCode(9 + extra/2, 7);
366 writeCode(extra, 1);
367 }
368 writeCode(bestDistance - 1, 5); // no extra bits, because we only searched 4 back
369 writeIndex += bestLength;
370 } else { // literals in DEFLATE's fixed Huffman
371 uint32_t c = block[writeIndex];
372 if (c <= 143) {
373 writeCode(c + 48, 8);
374 } else {
375 writeCode(c + 0x100, 9);
376 }
377 ++writeIndex;
378 }
379 }
380 writeCode(0, 7); // end-of-block code
381 }
