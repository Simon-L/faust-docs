42template <typename C> struct mapping
43{
44 const C fMinOut;
45 const C fMaxOut;
46 mapping(C omin, C omax) : fMinOut(omin), fMaxOut(omax) {}
47 C clip (C x) { return (x < fMinOut) ? fMinOut : (x > fMaxOut) ? fMaxOut : x; }
48};

54template <typename C> class FaustNode : public MessageDriven, public uiTypedItemReal<C>
55{
56 mapping<C> fMapping;
57 RootNode* fRoot;
58 bool fInput; // true for input nodes (slider, button...)
59
60 //---------------------------------------------------------------------
61 // Warning !!!
62 // The cast (C*)fZone is necessary because the real size allocated is
63 // only known at execution time. When the library is compiled, fZone is
64 // uniquely defined by FAUSTFLOAT.
65 //---------------------------------------------------------------------
66 bool store(C val) { *(C*)this->fZone = fMapping.clip(val); return true; }
67 void sendOSC() const;
68
69 protected:
70 FaustNode(RootNode* root, const char *name, C* zone, C init, C min, C max, const char* prefix, GUI* ui, bool initZone, bool input)
71 : MessageDriven(name, prefix), uiTypedItemReal<C>(ui, zone), fMapping(min, max), fRoot(root), fInput(input)
72 {
73 if (initZone) {
74 *zone = init;
75 }
76 }
77
78 virtual ~FaustNode() {}
79
80 public:
81 typedef SMARTP<FaustNode<C> > SFaustNode;
82 static SFaustNode create(RootNode* root, const char* name, C* zone, C init, C min, C max, const char* prefix, GUI* ui, bool initZone, bool input)
83 {
84 SFaustNode node = new FaustNode(root, name, zone, init, min, max, prefix, ui, initZone, input);
85 /*
86 Since FaustNode is a subclass of uiItem, the pointer will also be kept in the GUI class, and it's desallocation will be done there.
87 So we don't want to have smartpointer logic desallocate it and we increment the refcount.
88 */
89 node->addReference();
90 return node;
91 }

93 bool accept(const Message* msg);
94 void get(unsigned long ipdest) const;
95 virtual void reflectZone() { sendOSC(); this->fCache = *this->fZone; }
96};
