130 // -- soundfiles
131 virtual void addSoundfile(const char* label, const char* url, Soundfile** sf_zone)
132 {
133 const char* saved_url = url; // 'url' is consumed by parseMenuList2
134 std::vector<std::string> file_name_list;
135
136 bool menu = parseMenuList2(url, file_name_list, true);
137 // If not a list, we have as single file
138 if (!menu) { file_name_list.push_back(saved_url); }
139
140 // Parse the possible list
141 std::string saved_url_real = std::string(saved_url) + "_" + std::to_string(fIsDouble); // fIsDouble is used in the key
142 if (fSoundfileMap.find(saved_url_real) == fSoundfileMap.end()) {
143 // Check all files and get their complete path
144 std::vector<std::string> path_name_list = fSoundReader->checkFiles(fSoundfileDir, file_name_list);
145 // Read them and create the Soundfile
146 Soundfile* sound_file = fSoundReader->createSoundfile(path_name_list, MAX_CHAN, fIsDouble);
147 if (sound_file) {
148 fSoundfileMap[saved_url_real] = std::shared_ptr<Soundfile>(sound_file);
149 } else {
150 // If failure, use 'defaultsound'
151 std::cerr << "addSoundfile : soundfile for " << saved_url << " cannot be created !" << std::endl;
152 *sf_zone = defaultsound;
153 return;
154 }
155 }
156
157 // Get the soundfile pointer
158 *sf_zone = fSoundfileMap[saved_url_real].get();
159 }

167 static std::string getBinaryPath()
168 {
169 std::string bundle_path_str;
170 #if defined(__APPLE__) && !defined(__VCVRACK__) && !defined(JUCE_32BIT) && !defined(JUCE_64BIT)
171 CFURLRef bundle_ref = CFBundleCopyBundleURL(CFBundleGetMainBundle());
172 if (!bundle_ref) { std::cerr << "getBinaryPath CFBundleCopyBundleURL error\n"; return ""; }
173
174 UInt8 bundle_path[1024];
175 if (CFURLGetFileSystemRepresentation(bundle_ref, true, bundle_path, 1024)) {
176 bundle_path_str = std::string((char*)bundle_path);
177 } else {
178 std::cerr << "getBinaryPath CFURLGetFileSystemRepresentation error\n";
179 }
180 #endif
181 #ifdef ANDROID_DRIVER
182 bundle_path_str = "/data/data/__CURRENT_ANDROID_PACKAGE__/files";
183 #endif
184 return bundle_path_str;
185 }

226 static std::vector<std::string> getSoundfilePaths(dsp* dsp)
227 {
228 // Analyse 'soundfiles' metadata to extract the list of URLs.
229 struct SoundfilesMeta : Meta
230 {
231 std::vector<std::string> fURL;
232 void declare(const char* key, const char* value)
233 {
234 if (std::string(key) == "soundfiles") {
235 std::stringstream ss(value);
236 std::string item;
237 // Use getline with ';' as the delimiter to split the string
238 while (getline(ss, item, ';')) { fURL.push_back(item); }
239 }
240 }
241 };
242
243 // Use bundle path
244 SoundfilesMeta sf;
245 dsp->metadata(&sf);
246 return sf.fURL;
247 };
