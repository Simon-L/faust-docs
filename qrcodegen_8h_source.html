79/*
80 * A segment of user/application data that a QR Code symbol can convey.
81 * Each segment has a mode, a character count, and character/general data that is
82 * already encoded as a sequence of bits. The maximum allowed bit length is 32767,
83 * because even the largest QR Code (version 40) has only 31329 modules.
84 */
85struct qrcodegen_Segment {
86 // The mode indicator for this segment.
87 enum qrcodegen_Mode mode;
88
89 // The length of this segment's unencoded data. Always in the range [0, 32767].
90 // For numeric, alphanumeric, and kanji modes, this measures in Unicode code points.
91 // For byte mode, this measures in bytes (raw binary data, text in UTF-8, or other encodings).
92 // For ECI mode, this is always zero.
93 int numChars;
94
95 // The data bits of this segment, packed in bitwise big endian.
96 // Can be null if the bit length is zero.
97 uint8_t *data;
98
99 // The number of valid data bits used in the buffer. Requires
100 // 0 <= bitLength <= 32767, and bitLength <= (capacity of data array) * 8.
101 int bitLength;
102};

106// The minimum and maximum defined QR Code version numbers for Model 2.
107#define qrcodegen_VERSION_MIN 1
108#define qrcodegen_VERSION_MAX 40
109
110// Calculates the number of bytes needed to store any QR Code up to and including the given version number,
111// as a compile-time constant. For example, 'uint8_t buffer[qrcodegen_BUFFER_LEN_FOR_VERSION(25)];'
112// can store any single QR Code from version 1 to 25, inclusive.
113// Requires qrcodegen_VERSION_MIN <= n <= qrcodegen_VERSION_MAX.
114#define qrcodegen_BUFFER_LEN_FOR_VERSION(n) ((((n) * 4 + 17) * ((n) * 4 + 17) + 7) / 8 + 1)
115
116// The worst-case number of bytes needed to store one QR Code, up to and including
117// version 40. This value equals 3918, which is just under 4 kilobytes.
118// Use this more convenient value to avoid calculating tighter memory bounds for buffers.
119#define qrcodegen_BUFFER_LEN_MAX qrcodegen_BUFFER_LEN_FOR_VERSION(qrcodegen_VERSION_MAX)

127/*
128 * Encodes the given text string to a QR Code symbol, returning true if encoding succeeded.
129 * If the data is too long to fit in any version in the given range
130 * at the given ECC level, then false is returned.
131 * - The input text must be encoded in UTF-8 and contain no NULs.
132 * - The variables ecl and mask must correspond to enum constant values.
133 * - Requires 1 <= minVersion <= maxVersion <= 40.
134 * - The arrays tempBuffer and qrcode must each have a length
135 * of at least qrcodegen_BUFFER_LEN_FOR_VERSION(maxVersion).
136 * - After the function returns, tempBuffer contains no useful data.
137 * - If successful, the resulting QR Code may use numeric,
138 * alphanumeric, or byte mode to encode the text.
139 * - In the most optimistic case, a QR Code at version 40 with low ECC
140 * can hold any UTF-8 string up to 2953 bytes, or any alphanumeric string
141 * up to 4296 characters, or any digit string up to 7089 characters.
142 * These numbers represent the hard upper limit of the QR Code standard.
143 * - Please consult the QR Code specification for information on
144 * data capacities per version, ECC level, and text encoding mode.
145 */
146static bool qrcodegen_encodeText(const char *text, uint8_t tempBuffer[], uint8_t qrcode[],
147 enum qrcodegen_Ecc ecl, int minVersion, int maxVersion, enum qrcodegen_Mask mask, bool boostEcl);
148
149/*
150 * Encodes the given binary data to a QR Code symbol, returning true if encoding succeeded.
151 * If the data is too long to fit in any version in the given range
152 * at the given ECC level, then false is returned.
153 * - The input array range dataAndTemp[0 : dataLen] should normally be
154 * valid UTF-8 text, but is not required by the QR Code standard.
155 * - The variables ecl and mask must correspond to enum constant values.
156 * - Requires 1 <= minVersion <= maxVersion <= 40.
157 * - The arrays dataAndTemp and qrcode must each have a length
158 * of at least qrcodegen_BUFFER_LEN_FOR_VERSION(maxVersion).
159 * - After the function returns, the contents of dataAndTemp may have changed,
160 * and does not represent useful data anymore.
161 * - If successful, the resulting QR Code will use byte mode to encode the data.
162 * - In the most optimistic case, a QR Code at version 40 with low ECC can hold any byte
163 * sequence up to length 2953. This is the hard upper limit of the QR Code standard.
164 * - Please consult the QR Code specification for information on
165 * data capacities per version, ECC level, and text encoding mode.
166 */
167static bool qrcodegen_encodeBinary(uint8_t dataAndTemp[], size_t dataLen, uint8_t qrcode[],
168 enum qrcodegen_Ecc ecl, int minVersion, int maxVersion, enum qrcodegen_Mask mask, bool boostEcl);

180/*
181 * Returns the number of bytes (uint8_t) needed for the data buffer of a segment
182 * containing the given number of characters using the given mode. Notes:
183 * - Returns SIZE_MAX on failure, i.e. numChars > INT16_MAX or
184 * the number of needed bits exceeds INT16_MAX (i.e. 32767).
185 * - Otherwise, all valid results are in the range [0, ceil(INT16_MAX / 8)], i.e. at most 4096.
186 * - It is okay for the user to allocate more bytes for the buffer than needed.
187 * - For byte mode, numChars measures the number of bytes, not Unicode code points.
188 * - For ECI mode, numChars must be 0, and the worst-case number of bytes is returned.
189 * An actual ECI segment can have shorter data. For non-ECI modes, the result is exact.
190 */
191static size_t qrcodegen_calcSegmentBufferSize(enum qrcodegen_Mode mode, size_t numChars);

216/*
217 * Renders a QR Code symbol representing the given data segments at the given error correction
218 * level or higher. The smallest possible QR Code version is automatically chosen for the output.
219 * Returns true if QR Code creation succeeded, or false if the data is too long to fit in any version.
220 * This function allows the user to create a custom sequence of segments that switches
221 * between modes (such as alphanumeric and binary) to encode text more efficiently.
222 * This function is considered to be lower level than simply encoding text or binary data.
223 * To save memory, the segments' data buffers can alias/overlap tempBuffer, and will
224 * result in them being clobbered, but the QR Code output will still be correct.
225 * But the qrcode array must not overlap tempBuffer or any segment's data buffer.
226 */
227static bool qrcodegen_encodeSegments(const struct qrcodegen_Segment segs[], size_t len,
228 enum qrcodegen_Ecc ecl, uint8_t tempBuffer[], uint8_t qrcode[]);
