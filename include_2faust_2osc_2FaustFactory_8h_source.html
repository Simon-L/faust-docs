82template <typename C> void FaustFactory::addnode(const char* label, C* zone, C init, C min, C max, bool initZone, bool input)
83{
84 SMessageDriven top;
85 if (fNodes.size()) top = fNodes.top();
86 if (top) {
87 std::string prefix = top->getOSCAddress();
