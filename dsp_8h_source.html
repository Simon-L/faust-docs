291class FAUST_API ScopedNoDenormals {
292
293 private:
294
295 intptr_t fpsr = 0;
296
297 void setFpStatusRegister(intptr_t fpsr_aux) noexcept
298 {
299 #if defined (__arm64__) || defined (__aarch64__)
300 asm volatile("msr fpcr, %0" : : "ri" (fpsr_aux));
301 #elif defined (__SSE__)
302 // The volatile keyword here is needed to workaround a bug in AppleClang 13.0
303 // which aggressively optimises away the variable otherwise
304 volatile uint32_t fpsr_w = static_cast<uint32_t>(fpsr_aux);
305 _mm_setcsr(fpsr_w);
306 #endif
307 }
308
309 void getFpStatusRegister() noexcept
310 {
311 #if defined (__arm64__) || defined (__aarch64__)
312 asm volatile("mrs %0, fpcr" : "=r" (fpsr));
313 #elif defined (__SSE__)
314 fpsr = static_cast<intptr_t>(_mm_getcsr());
315 #endif
316 }
317
318 public:
319
320 ScopedNoDenormals() noexcept
321 {
322 #if defined (__arm64__) || defined (__aarch64__)
323 intptr_t mask = (1 << 24 /* FZ */);
324 #elif defined (__SSE__)
325 #if defined (__SSE2__)
326 intptr_t mask = 0x8040;
327 #else
328 intptr_t mask = 0x8000;
329 #endif
330 #else
331 intptr_t mask = 0x0000;
332 #endif
333 getFpStatusRegister();
334 setFpStatusRegister(fpsr | mask);
335 }
336
337 ~ScopedNoDenormals() noexcept
338 {
339 setFpStatusRegister(fpsr);
340 }
341
342};
