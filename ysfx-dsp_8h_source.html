55 void updateInputsControl()
56 {
57 for (int i = 0; i < fInputsValue.size(); i++) {
58 ysfx_slider_set_value(fDSP, i, fInputsValue[i]->fValue);
59 }
60 }

62 void updateOutputsControl()
63 {
64 for (int i = 0; i < fOutputsValue.size(); i++) {
65 fOutputsValue[i]->fValue = ysfx_slider_get_value(fDSP, i);
66 }
67 }

71 ysfx_dsp(const char* filename)
72 {
73 fConfig = ysfx_config_new();
74 fDSP = ysfx_new(fConfig);
75 if (!ysfx_load_file(fDSP, filename, 0)) {
76 std::cerr << "File not found:" << filename << endl;
77 throw std::bad_alloc();
78 }
79 if (!ysfx_compile(fDSP, 0)) {
80 cout << "Cannot compile:" << filename << endl;
81 throw std::bad_alloc();
82 }
83 fFileName = filename;
84 }
85
86 virtual ~ysfx_dsp()
87 {
88 ysfx_config_free(fConfig);
89 ysfx_free(fDSP);
90 for (int i = 0; i < fInputsValue.size(); i++) {
91 delete fInputsValue[i];
92 }
93 for (int i = 0; i < fOutputsValue.size(); i++) {
94 delete fOutputsValue[i];
95 }
96 }

108 void buildUserInterface(UI* ui_interface)
109 {
110 int i = 0;
111 ui_interface->openVerticalBox("JSFX");
112 while (ysfx_slider_exists(fDSP, i)) {
113 ysfx_slider_range_t ra;
114 ysfx_slider_get_range(fDSP, i, &ra);
115 fInputsValue.push_back(new ControlValue(ra.def));
116 string name = ysfx_slider_get_name(fDSP, i);
117 if (startWith(name, "button")) {
118 ui_interface->addButton(ysfx_slider_get_name(fDSP, i), &fInputsValue[i]->fValue);
119 } else if (startWith(name, "checkbox")) {
120 ui_interface->addCheckbox(ysfx_slider_get_name(fDSP, i), &fInputsValue[i]->fValue);
121 } else if (startWith(name, "hslider")) {
122 ui_interface->addHorizontalSlider(ysfx_slider_get_name(fDSP, i),
123 &fInputsValue[i]->fValue,
124 ra.def,
125 ra.min,
126 ra.max,
127 ra.inc);
128 } else if (startWith(name, "vslider")) {
129 ui_interface->addVorizontalSlider(ysfx_slider_get_name(fDSP, i),
130 &fInputsValue[i]->fValue,
131 ra.def,
132 ra.min,
133 ra.max,
134 ra.inc);
135 } else if (startWith(name, "nentry")) {
136 ui_interface->addNumEntry(ysfx_slider_get_name(fDSP, i),
137 &fInputsValue[i]->fValue,
138 ra.def,
139 ra.min,
140 ra.max,
141 ra.inc);
142 }
143 i++;
144 }
145 ui_interface->closeBox();
146 }

174 void compute(int count, FAUSTFLOAT** input, FAUSTFLOAT** output)
175 {
176 updateInputsControl();
177 if (sizeof(REAL) == sizeof(float)) {
178 ysfx_process_float(fDSP, (const float **)input, (float **)output, getNumInputs(), getNumOutputs(), count);
179 } else {
180 ysfx_process_double(fDSP, (const double **)input, (double **)output, getNumInputs(), getNumOutputs(), count);
181 }
182 updateOutputsControl();
183 }
