310 // Override this method if your AU processes multiple output busses completely independently --
311 // you'll want to just call Render without the NeedsToRender check.
312 // Otherwise, override Render().
313 //
314 // N.B. Implementations of this method can assume that the output's buffer list has already been
315 // prepared and access it with GetOutput(inBusNumber)->GetBufferList() instead of
316 // GetOutput(inBusNumber)->PrepareBuffer(nFrames) -- if PrepareBuffer is called, a
317 // copy may occur after rendering.
319 virtual OSStatus RenderBus( AudioUnitRenderActionFlags & ioActionFlags,
320 const AudioTimeStamp & inTimeStamp,
321 UInt32 inBusNumber,
322 UInt32 inNumberFrames)
323 {
324 if (NeedsToRender(inTimeStamp))
325 return Render(ioActionFlags, inTimeStamp, inNumberFrames);
326 return noErr; // was presumably already rendered via another bus
327 }
328
329 // N.B. For a unit with only one output bus, it can assume in its implementation of this
330 // method that the output's buffer list has already been prepared and access it with
331 // GetOutput(0)->GetBufferList() instead of GetOutput(0)->PrepareBuffer(nFrames)
332 // -- if PrepareBuffer is called, a copy may occur after rendering.
334 virtual OSStatus Render( AudioUnitRenderActionFlags & ioActionFlags,
335 const AudioTimeStamp & inTimeStamp,
336 UInt32 inNumberFrames)
337 {
338 return noErr;
339 }

776private:
778 // shared between Render and RenderSlice, inlined to minimize function call overhead
779 OSStatus DoRenderBus( AudioUnitRenderActionFlags & ioActionFlags,
780 const AudioTimeStamp & inTimeStamp,
781 UInt32 inBusNumber,
782 AUOutputElement * theOutput,
783 UInt32 inNumberFrames,
784 AudioBufferList & ioData)
785 {
786 if (ioData.mBuffers[0].mData == NULL || (theOutput->WillAllocateBuffer() && Outputs().GetNumberOfElements() > 1))
787 // will render into cache buffer
788 theOutput->PrepareBuffer(inNumberFrames);
789 else
790 // will render into caller's buffer
791 theOutput->SetBufferList(ioData);
792 OSStatus result = RenderBus(ioActionFlags, inTimeStamp, inBusNumber, inNumberFrames);
793 if (result == noErr) {
794 if (ioData.mBuffers[0].mData == NULL) {
795 theOutput->CopyBufferListTo(ioData);
796 AUTRACE(kCATrace_AUBaseDoRenderBus, mComponentInstance, inNumberFrames, (intptr_t)theOutput->GetBufferList().mBuffers[0].mData, 0, *(UInt32 *)ioData.mBuffers[0].mData);
797 } else {
798 theOutput->CopyBufferContentsTo(ioData);
799 AUTRACE(kCATrace_AUBaseDoRenderBus, mComponentInstance, inNumberFrames, (intptr_t)theOutput->GetBufferList().mBuffers[0].mData, (intptr_t)ioData.mBuffers[0].mData, *(UInt32 *)ioData.mBuffers[0].mData);
800 theOutput->InvalidateBufferList();
801 }
802 }
803 return result;
804 }
