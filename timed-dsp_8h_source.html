132class timed_dsp : public decorator_dsp {
133
134 protected:
135
136 double fDateUsec; // Compute call date in usec
137 double fOffsetUsec; // Compute call offset in usec
138 bool fFirstCallback;
139 ZoneUI fZoneUI;
140
141 FAUSTFLOAT** fInputsSlice;
142 FAUSTFLOAT** fOutputsSlice;
143
144 void computeSlice(int offset, int slice, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
145 {
146 if (slice > 0) {
147 for (int chan = 0; chan < fDSP->getNumInputs(); chan++) {
148 fInputsSlice[chan] = &(inputs[chan][offset]);
149 }
150 for (int chan = 0; chan < fDSP->getNumOutputs(); chan++) {
151 fOutputsSlice[chan] = &(outputs[chan][offset]);
152 }
153 fDSP->compute(slice, fInputsSlice, fOutputsSlice);
154 }
155 }
156
157 double convertUsecToSample(double usec)
158 {
159 return std::max<double>(0., (double(getSampleRate()) * (usec - fDateUsec)) / 1000000.);
160 }
161
162 ztimedmap::iterator getNextControl(DatedControl& res)
163 {
164 DatedControl date1(DBL_MAX, 0);
165 ztimedmap::iterator it1, it2 = GUI::gTimedZoneMap.end();
166 std::set<FAUSTFLOAT*>::iterator it3;
167
168 // Find date of next audio slice to compute
169 for (it3 = fZoneUI.fZoneSet.begin(); it3 != fZoneUI.fZoneSet.end(); it3++) {
170 // If value list is not empty, get the date and keep the minimal one
171 it1 = GUI::gTimedZoneMap.find(*it3);
172 if (it1 != GUI::gTimedZoneMap.end()) { // Check if zone still in global GUI::gTimedZoneMap (since MidiUI may have been desallocated)
173 DatedControl date2;
174 if (ringbuffer_peek((*it1).second, (char*)&date2, sizeof(DatedControl)) == sizeof(DatedControl)
175 && date2.fDate < date1.fDate) {
176 it2 = it1;
177 date1 = date2;
178 }
179 }
180 }
181
182 res = date1;
183 return it2;
184 }
185
186 virtual void computeAux(int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs, bool convert_ts)
187 {
188 int slice, offset = 0;
189 ztimedmap::iterator it;
190 DatedControl next_control;
191
192 // Do audio computation "slice" by "slice"
193 while ((it = getNextControl(next_control)) != GUI::gTimedZoneMap.end()) {
194
195 // If needed, convert next_control in samples from begining of the buffer, possible moving to 0 (if negative)
196 if (convert_ts) {
197 next_control.fDate = convertUsecToSample(next_control.fDate);
198 }
199
200 // Compute audio slice
201 slice = int(next_control.fDate) - offset;
202 computeSlice(offset, slice, inputs, outputs);
203 offset += slice;
204
205 // Update control
206 ringbuffer_t* control_values = (*it).second;
207 *((*it).first) = next_control.fValue;
208
209 // Move ringbuffer pointer
210 ringbuffer_read_advance(control_values, sizeof(DatedControl));
211 }
212
213 // Compute last audio slice
214 slice = count - offset;
215 computeSlice(offset, slice, inputs, outputs);
216 }
217
218 public:
219
220 timed_dsp(dsp* dsp):decorator_dsp(dsp), fDateUsec(0), fOffsetUsec(0), fFirstCallback(true)
221 {
222 fInputsSlice = new FAUSTFLOAT*[dsp->getNumInputs()];
223 fOutputsSlice = new FAUSTFLOAT*[dsp->getNumOutputs()];
224 }
225 virtual ~timed_dsp()
226 {
227 delete [] fInputsSlice;
228 delete [] fOutputsSlice;
229 }
230

233 fDSP->init(sample_rate);
234 }
235
236 virtual void buildUserInterface(UI* ui_interface)
237 {
238 fDSP->buildUserInterface(ui_interface);

240 fDSP->buildUserInterface(&fZoneUI);
241 }
242
243 virtual timed_dsp* clone()
244 {
245 return new timed_dsp(fDSP->clone());
246 }
247
248 // Default method take a timestamp at 'compute' call time
249 virtual void compute(int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
250 {
251 compute(::GetCurrentTimeInUsec(), count, inputs, outputs);
252 }
253
254 virtual void compute(double date_usec, int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
255 {
256 if (date_usec == -1) {
257 // Timestamp is expressed in frames
258 computeAux(count, inputs, outputs, false);
259 } else {
260 // Save the timestamp offset in the first callback
261 if (fFirstCallback) {
262 fFirstCallback = false;
263 double current_date_usec = ::GetCurrentTimeInUsec();
264 fDateUsec = current_date_usec;
265 fOffsetUsec = current_date_usec - date_usec;
266 }
267
268 // RtMidi mode: timestamp must be converted in frames
269 computeAux(count, inputs, outputs, true);
270
271 // Keep call date
272 fDateUsec = date_usec + fOffsetUsec;
273 }
274 }
