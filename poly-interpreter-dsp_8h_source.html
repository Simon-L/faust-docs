35struct interpreter_dsp_poly_factory : public dsp_poly_factory {
36
37 interpreter_dsp_poly_factory(const std::string& name_app,
38 const std::string& dsp_content,
39 int argc, const char* argv[],
40 std::string& error_msg)
41 {
42 fProcessFactory = createInterpreterDSPFactoryFromString(name_app, dsp_content, argc, argv, error_msg);
43 if (fProcessFactory) {
44 fEffectFactory = createInterpreterDSPFactoryFromString(name_app, getEffectCode(dsp_content), argc, argv, error_msg);
45 if (!fEffectFactory) {
46 std::cerr << "interpreter_dsp_poly_factory : fEffectFactory " << error_msg;
47 // The error message is not really needed...
48 error_msg = "";
49 }
50 } else {
51 std::cerr << "interpreter_dsp_poly_factory : fProcessFactory " << error_msg;
52 throw std::bad_alloc();
53 }
54 }
55
56 interpreter_dsp_poly_factory(interpreter_dsp_factory* process_factory, interpreter_dsp_factory* effect_factory)
57 :dsp_poly_factory(process_factory, effect_factory)
58 {}
59
60 virtual ~interpreter_dsp_poly_factory()
61 {
62 deleteInterpreterDSPFactory(static_cast<interpreter_dsp_factory*>(fProcessFactory));
63 deleteInterpreterDSPFactory(static_cast<interpreter_dsp_factory*>(fEffectFactory));
64 }
65
66};

115static dsp_poly_factory* readInterpreterPolyDSPFactoryFromMachineFile(const std::string& bit_code_path, std::string& error_msg)
116{
117 std::string process_path = bit_code_path + "_bitcode_process.fbc";
118 std::string effect_path = bit_code_path + "_bicode_effect.fbc";
119 interpreter_dsp_factory* process_factory = readInterpreterDSPFactoryFromBitcodeFile(process_path, error_msg);
120 interpreter_dsp_factory* effect_factory = readInterpreterDSPFactoryFromBitcodeFile(effect_path, error_msg);
121 if (process_factory) {
122 return new interpreter_dsp_poly_factory(process_factory, effect_factory);
123 } else {
124 interpreter_dsp_factory* process_factory = readInterpreterDSPFactoryFromBitcodeFile(bit_code_path, error_msg);
125 return (process_factory) ? new interpreter_dsp_poly_factory(process_factory, NULL) : NULL;
126 }
127}

136static void writeInterpreterPolyDSPFactoryToMachineFile(dsp_poly_factory* factory, const std::string& bit_code_path)
137{
138 std::string process_path = bit_code_path + "_bitcode_process.fbc";
139 if (factory->fEffectFactory) {
140 std::string effect_path = bit_code_path + "_bicode_effect.fbc";
141 writeInterpreterDSPFactoryToBitcodeFile(static_cast<interpreter_dsp_factory*>(factory->fEffectFactory), effect_path);
142 }
143 writeInterpreterDSPFactoryToBitcodeFile(static_cast<interpreter_dsp_factory*>(factory->fProcessFactory), process_path);
144}
