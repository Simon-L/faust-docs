100 // Add a generic parameter
101 virtual void addParameter(const char* label,
102 FAUSTFLOAT* zone,
103 FAUSTFLOAT init,
104 FAUSTFLOAT min,
105 FAUSTFLOAT max,
106 FAUSTFLOAT step,
107 ItemType type)
108 {
109 std::string path = buildPath(label);
110 fFullPaths.push_back(path);
111
112 // handle scale metadata
113 ValueConverter* converter = nullptr;
114 switch (fCurrentScale) {
115 case kLin:
116 converter = new LinearValueConverter(0, 1, min, max);
117 break;
118 case kLog:
119 converter = new LogValueConverter(0, 1, min, max);
120 break;
121 case kExp:
122 converter = new ExpValueConverter(0, 1, min, max);
123 break;
124 }
125 fCurrentScale = kLin;
126
127 fItems.push_back(Item(label, "", path, converter, zone, init, min, max, step, type));
128
129 if (fCurrentAcc.size() > 0 && fCurrentGyr.size() > 0) {
130 fprintf(stderr, "warning : 'acc' and 'gyr' metadata used for the same %s parameter !!\n", label);
131 }
132
133 // handle acc metadata "...[acc : <axe> <curve> <amin> <amid> <amax>]..."
134 if (fCurrentAcc.size() > 0) {
135 std::istringstream iss(fCurrentAcc);
136 int axe, curve;
137 double amin, amid, amax;
138 iss >> axe >> curve >> amin >> amid >> amax;
139
140 if ((0 <= axe) && (axe < 3) &&
141 (0 <= curve) && (curve < 4) &&
142 (amin < amax) && (amin <= amid) && (amid <= amax))
143 {
144 fAcc[axe].push_back(new CurveZoneControl(zone, curve, amin, amid, amax, min, init, max));
145 } else {
146 fprintf(stderr, "incorrect acc metadata : %s \n", fCurrentAcc.c_str());
147 }
148 fCurrentAcc = "";
149 }
150
151 // handle gyr metadata "...[gyr : <axe> <curve> <amin> <amid> <amax>]..."
152 if (fCurrentGyr.size() > 0) {
153 std::istringstream iss(fCurrentGyr);
154 int axe, curve;
155 double amin, amid, amax;
156 iss >> axe >> curve >> amin >> amid >> amax;
157
158 if ((0 <= axe) && (axe < 3) &&
159 (0 <= curve) && (curve < 4) &&
160 (amin < amax) && (amin <= amid) && (amid <= amax))
161 {
162 fGyr[axe].push_back(new CurveZoneControl(zone, curve, amin, amid, amax, min, init, max));
163 } else {
164 fprintf(stderr, "incorrect gyr metadata : %s \n", fCurrentGyr.c_str());
165 }
166 fCurrentGyr = "";
167 }
168
169 // handle screencolor metadata "...[screencolor:red|green|blue|white]..."
170 if (fCurrentColor.size() > 0) {
171 if ((fCurrentColor == "red") && (fRedReader == nullptr)) {
172 fRedReader = new ZoneReader(zone, min, max);
173 fHasScreenControl = true;
174 } else if ((fCurrentColor == "green") && (fGreenReader == nullptr)) {
175 fGreenReader = new ZoneReader(zone, min, max);
176 fHasScreenControl = true;
177 } else if ((fCurrentColor == "blue") && (fBlueReader == nullptr)) {
178 fBlueReader = new ZoneReader(zone, min, max);
179 fHasScreenControl = true;
180 } else if ((fCurrentColor == "white") && (fRedReader == nullptr) && (fGreenReader == nullptr) && (fBlueReader == nullptr)) {
181 fRedReader = new ZoneReader(zone, min, max);
182 fGreenReader = new ZoneReader(zone, min, max);
183 fBlueReader = new ZoneReader(zone, min, max);
184 fHasScreenControl = true;
185 } else {
186 fprintf(stderr, "incorrect screencolor metadata : %s \n", fCurrentColor.c_str());
187 }
188 }
189 fCurrentColor = "";
190
191 fMetaData.push_back(fCurrentMetadata);
192 fCurrentMetadata.clear();
193 }

195 int getZoneIndex(std::vector<ZoneControl*>* table, int p, int val)
196 {
197 FAUSTFLOAT* zone = fItems[uint(p)].fZone;
198 for (size_t i = 0; i < table[val].size(); i++) {
199 if (zone == table[val][i]->getZone()) return int(i);
200 }
201 return -1;
202 }

204 void setConverter(std::vector<ZoneControl*>* table, int p, int val, int curve, double amin, double amid, double amax)
205 {
206 int id1 = getZoneIndex(table, p, 0);
207 int id2 = getZoneIndex(table, p, 1);
208 int id3 = getZoneIndex(table, p, 2);
209
210 // Deactivates everywhere..
211 if (id1 != -1) table[0][uint(id1)]->setActive(false);
212 if (id2 != -1) table[1][uint(id2)]->setActive(false);
213 if (id3 != -1) table[2][uint(id3)]->setActive(false);
214
215 if (val == -1) { // Means: no more mapping...
216 // So stay all deactivated...
217 } else {
218 int id4 = getZoneIndex(table, p, val);
219 if (id4 != -1) {
220 // Reactivate the one we edit...
221 table[val][uint(id4)]->setMappingValues(curve, amin, amid, amax, fItems[uint(p)].fMin, fItems[uint(p)].fInit, fItems[uint(p)].fMax);
222 table[val][uint(id4)]->setActive(true);
223 } else {
224 // Allocate a new CurveZoneControl which is 'active' by default
225 FAUSTFLOAT* zone = fItems[uint(p)].fZone;
226 table[val].push_back(new CurveZoneControl(zone, curve, amin, amid, amax, fItems[uint(p)].fMin, fItems[uint(p)].fInit, fItems[uint(p)].fMax));
227 }
228 }
229 }

231 void getConverter(std::vector<ZoneControl*>* table, int p, int& val, int& curve, double& amin, double& amid, double& amax)
232 {
233 int id1 = getZoneIndex(table, p, 0);
234 int id2 = getZoneIndex(table, p, 1);
235 int id3 = getZoneIndex(table, p, 2);
236
237 if (id1 != -1) {
238 val = 0;
239 curve = table[val][uint(id1)]->getCurve();
240 table[val][uint(id1)]->getMappingValues(amin, amid, amax);
241 } else if (id2 != -1) {
242 val = 1;
243 curve = table[val][uint(id2)]->getCurve();
244 table[val][uint(id2)]->getMappingValues(amin, amid, amax);
245 } else if (id3 != -1) {
246 val = 2;
247 curve = table[val][uint(id3)]->getCurve();
248 table[val][uint(id3)]->getMappingValues(amin, amid, amax);
249 } else {
250 val = -1; // No mapping
251 curve = 0;
252 amin = -100.;
253 amid = 0.;
254 amax = 100.;
255 }

342
343 if (strcmp(key, "scale") == 0) {

345 fCurrentScale = kLog;
346 } else if (strcmp(val, "exp") == 0) {
347 fCurrentScale = kExp;
348 } else {
349 fCurrentScale = kLin;
350 }
351 } else if (strcmp(key, "unit") == 0) {
352 fCurrentUnit = val;
353 } else if (strcmp(key, "acc") == 0) {
354 fCurrentAcc = val;
355 } else if (strcmp(key, "gyr") == 0) {
356 fCurrentGyr = val;
357 } else if (strcmp(key, "screencolor") == 0) {
358 fCurrentColor = val; // val = "red", "green", "blue" or "white"
359 } else if (strcmp(key, "tooltip") == 0) {
360 fCurrentTooltip = val;
361 }
