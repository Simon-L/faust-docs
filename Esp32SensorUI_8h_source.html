40// To be used with Gramophone for accelerometer, gyroscope and magnetometer
41class Esp32SensorUI : public APIUI
42{
43 private:
44
45 TaskHandle_t fProcessHandle;
46 MPU9250 fMPU9250;
47 bool fHasAcc, fHasGyr, fHasMag;
48
49 void sensor()
50 {
51 while (true) {
52
53 if (fHasAcc) {
54 fMPU9250.accelUpdate();
55 propagateAcc(0, fMPU9250.accelX() * ONE_G);
56 propagateAcc(1, fMPU9250.accelY() * ONE_G);
57 propagateAcc(2, fMPU9250.accelZ() * ONE_G);
58 /*
59 std::cout << "======================" << std::endl;
60 std::cout << "accelX " << fMPU9250.accelX() << std::endl;
61 std::cout << "accelY " << fMPU9250.accelX() << std::endl;
62 std::cout << "accelZ " << fMPU9250.accelZ() << std::endl;
63 */
64 }
65
66 if (fHasGyr) {
67 fMPU9250.gyroUpdate();
68 propagateGyr(0, fMPU9250.gyroX());
69 propagateGyr(1, fMPU9250.gyroY());
70 propagateGyr(2, fMPU9250.gyroZ());
71 /*
72 std::cout << "======================" << std::endl;
73 std::cout << "gyroX " << fMPU9250.gyroX() << std::endl;
74 std::cout << "gyroY " << fMPU9250.gyroY() << std::endl;
75 std::cout << "gyroZ " << fMPU9250.gyroZ() << std::endl;
76 */
77 }
78
79 /*
80 if (fHasMag) {
81 fMPU9250.magUpdate();
82 propagateMag(0, fMPU9250.magX());
83 propagateMag(1, fMPU9250.magY());
84 propagateMag(2, fMPU9250.magZ());
85 propagateMag(2, fMPU9250.magHorizDirection());
86 }
87 */
88
89 vTaskDelay(5 / portTICK_PERIOD_MS);
90 }
91 }
92
93 static void sensorHandler(void* arg)
94 {
95 static_cast<Esp32SensorUI*>(arg)->sensor();
96 }
97
98 public:
99
100 Esp32SensorUI():fProcessHandle(nullptr),
101 fMPU9250(MPU9250_ADDRESS_AD0_HIGH),
102 fHasAcc(false),
103 fHasGyr(false),
104 fHasMag(false)
105 {}
106
107 bool start()
108 {
109 fHasAcc = (getAccCount(0) > 0) || (getAccCount(1) > 0) || (getAccCount(2) > 0);
110 fHasGyr = (getGyrCount(0) > 0) || (getGyrCount(1) > 0) || (getGyrCount(2) > 0);
111
112 fprintf(stdout, "fHasAcc %d\n", fHasAcc);
113 fprintf(stdout, "fHasGyr %d\n", fHasGyr);
114
115 if (fHasAcc) fMPU9250.beginAccel();
116 if (fHasGyr) fMPU9250.beginGyro();
117 if (fHasMag) fMPU9250.beginMag();
118 if (fHasAcc || fHasGyr || fHasMag) {
119 // Start Sensor receive task
120 return (xTaskCreatePinnedToCore(sensorHandler, "Faust Sensor Task", 10240, (void*)this, 5, &fProcessHandle, 1) == pdPASS);
121 } else {
122 return true;
123 }
124 }
125
126 void stop()
127 {
128 if (fProcessHandle != nullptr) {
129 vTaskDelete(fProcessHandle);
130 fProcessHandle = nullptr;
131 }
132 }
133
134};
