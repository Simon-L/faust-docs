39template <typename C> struct mapping
40{
41 const C fMinIn;
42 const C fMaxIn;
43 const C fMinOut;
44 const C fMaxOut;
45 const C fScale;
46
47 mapping(C imin, C imax, C omin, C omax) : fMinIn(imin), fMaxIn(imax),
48 fMinOut(omin), fMaxOut(omax),
49 fScale( (fMaxOut-fMinOut)/(fMaxIn-fMinIn) ) {}
50 C scale (C x) { C z = (x < fMinIn) ? fMinIn : (x > fMaxIn) ? fMaxIn : x; return fMinOut + (z - fMinIn) * fScale; }
51};

97 virtual bool accept(const Message* msg, std::vector<Message*>& outMsg)
98 {
99 if (msg->size() == 2) { // checks for the message parameters count
100 // messages with a param count other than 2 are rejected
101 std::string key;
102 if (msg->param(0, key) && (key == "value")) {
103 float val=0;
104 if (msg->param(1, val)) {
105 store(val); // accepts float values
106 }
107 get(outMsg);
108 return true;
109 }
110 }
111 return MessageDriven::accept(msg, outMsg);
112 }
