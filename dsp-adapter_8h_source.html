38// Adapts a DSP for a different number of inputs/outputs
39class dsp_adapter : public decorator_dsp {
40
41 private:
42
43 FAUSTFLOAT** fAdaptedInputs;
44 FAUSTFLOAT** fAdaptedOutputs;
45 int fHWInputs;
46 int fHWOutputs;
47 int fDSPInputs;
48 int fDSPOutputs;
49
50 int fBufferSize;
51 bool fDelete;
52
53 void adaptBuffers(FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
54 {
55 for (int i = 0; i < fHWInputs; i++) {
56 fAdaptedInputs[i] = inputs[i];
57 }
58 for (int i = 0; i < fHWOutputs; i++) {
59 fAdaptedOutputs[i] = outputs[i];
60 }
61 }
62
63 public:
64
65 dsp_adapter(dsp* dsp, int hw_inputs, int hw_outputs, int buffer_size, bool to_delete = true):decorator_dsp(dsp)
66 {
67 fHWInputs = hw_inputs;
68 fHWOutputs = hw_outputs;
69 fDSPInputs = dsp->getNumInputs();
70 fDSPOutputs = dsp->getNumOutputs();
71 fBufferSize = buffer_size;
72 fDelete = to_delete;
73
74 fAdaptedInputs = new FAUSTFLOAT*[std::max<int>(dsp->getNumInputs(), hw_inputs)];
75 for (int i = 0; i < dsp->getNumInputs() - fHWInputs; i++) {
76 fAdaptedInputs[i + fHWInputs] = new FAUSTFLOAT[buffer_size];
77 memset(fAdaptedInputs[i + fHWInputs], 0, sizeof(FAUSTFLOAT) * buffer_size);
78 }
79
80 fAdaptedOutputs = new FAUSTFLOAT*[std::max<int>(dsp->getNumOutputs(), hw_outputs)];
81 for (int i = 0; i < dsp->getNumOutputs() - fHWOutputs; i++) {
82 fAdaptedOutputs[i + fHWOutputs] = new FAUSTFLOAT[buffer_size];
83 memset(fAdaptedOutputs[i + fHWOutputs], 0, sizeof(FAUSTFLOAT) * buffer_size);
84 }
85 }
86
87 virtual ~dsp_adapter()
88 {
89 for (int i = 0; i < fDSPInputs - fHWInputs; i++) {
90 delete [] fAdaptedInputs[i + fHWInputs];
91 }
92 delete [] fAdaptedInputs;
93
94 for (int i = 0; i < fDSPOutputs - fHWOutputs; i++) {
95 delete [] fAdaptedOutputs[i + fHWOutputs];
96 }
97 delete [] fAdaptedOutputs;
98
99 // Decorator should not delete the decorated fDSP
100 if (!fDelete) fDSP = nullptr;
101 }
102
103 virtual int getNumInputs() { return fHWInputs; }
104 virtual int getNumOutputs() { return fHWOutputs; }
105
106 virtual dsp_adapter* clone() { return new dsp_adapter(fDSP->clone(), fHWInputs, fHWOutputs, fBufferSize); }
107
108 virtual void compute(double date_usec, int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
109 {
110 adaptBuffers(inputs, outputs);
111 fDSP->compute(date_usec, count, fAdaptedInputs, fAdaptedOutputs);
112 }
113
114 virtual void compute(int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
115 {
116 adaptBuffers(inputs, outputs);
117 fDSP->compute(count, fAdaptedInputs, fAdaptedOutputs);
118 }
119};

121// Adapts a DSP for a different sample size
122template <typename REAL_INT, typename REAL_EXT>
123class dsp_sample_adapter : public decorator_dsp {
124
125 private:
126
127 REAL_INT** fAdaptedInputs;
128 REAL_INT** fAdaptedOutputs;
129
130 void adaptInputBuffers(int count, FAUSTFLOAT** inputs)
131 {
132 for (int chan = 0; chan < fDSP->getNumInputs(); chan++) {
133 for (int frame = 0; frame < count; frame++) {
134 fAdaptedInputs[chan][frame] = REAL_INT(reinterpret_cast<REAL_EXT**>(inputs)[chan][frame]);
135 }
136 }
137 }
138
139 void adaptOutputsBuffers(int count, FAUSTFLOAT** outputs)
140 {
141 for (int chan = 0; chan < fDSP->getNumOutputs(); chan++) {
142 for (int frame = 0; frame < count; frame++) {
143 reinterpret_cast<REAL_EXT**>(outputs)[chan][frame] = REAL_EXT(fAdaptedOutputs[chan][frame]);
144 }
145 }
146 }
147
148 public:
149
150 dsp_sample_adapter(dsp* dsp):decorator_dsp(dsp)
151 {
152 fAdaptedInputs = new REAL_INT*[dsp->getNumInputs()];
153 for (int i = 0; i < dsp->getNumInputs(); i++) {
154 fAdaptedInputs[i] = new REAL_INT[4096];
155 }
156
157 fAdaptedOutputs = new REAL_INT*[dsp->getNumOutputs()];
158 for (int i = 0; i < dsp->getNumOutputs(); i++) {
159 fAdaptedOutputs[i] = new REAL_INT[4096];
160 }
161 }
162
163 virtual ~dsp_sample_adapter()
164 {
165 for (int i = 0; i < fDSP->getNumInputs(); i++) {
166 delete [] fAdaptedInputs[i];
167 }
168 delete [] fAdaptedInputs;
169
170 for (int i = 0; i < fDSP->getNumOutputs(); i++) {
171 delete [] fAdaptedOutputs[i];
172 }
173 delete [] fAdaptedOutputs;
174 }
175
176 virtual dsp_sample_adapter* clone() { return new dsp_sample_adapter(fDSP->clone()); }
177
178 virtual void compute(int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
179 {
180 assert(count <= 4096);
181 adaptInputBuffers(count, inputs);
182 // DSP base class uses FAUSTFLOAT** type, so reinterpret_cast has to be used even if the real DSP uses REAL_INT
183 fDSP->compute(count, reinterpret_cast<FAUSTFLOAT**>(fAdaptedInputs), reinterpret_cast<FAUSTFLOAT**>(fAdaptedOutputs));
184 adaptOutputsBuffers(count, outputs);
185 }
186
187 virtual void compute(double date_usec, int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
188 {
189 assert(count <= 4096);
190 adaptInputBuffers(count, inputs);
191 // DSP base class uses FAUSTFLOAT** type, so reinterpret_cast has to be used even if the real DSP uses REAL_INT
192 fDSP->compute(date_usec, count, reinterpret_cast<FAUSTFLOAT**>(fAdaptedInputs), reinterpret_cast<FAUSTFLOAT**>(fAdaptedOutputs));
193 adaptOutputsBuffers(count, outputs);
194 }
195};

221template <class fVslider0, int fVslider1, typename REAL>

223
224 REAL fVec0[2];
225 REAL fRec1[2];
226 REAL fRec0[3];
227
228 inline REAL LowPass3_faustpower2_f(REAL value)
229 {
230 return (value * value);
231 }
232
233 LowPass3()
234 {
235 for (int l0 = 0; (l0 < 2); l0 = (l0 + 1)) {
236 fVec0[l0] = 0.0;
237 }
238 for (int l1 = 0; (l1 < 2); l1 = (l1 + 1)) {
239 fRec1[l1] = 0.0;

246 inline void compute(int count, FAUSTFLOAT* input0, FAUSTFLOAT* output0)
247 {
248 // Computed at template specialization time
249 REAL fSlow0 = std::tan((3.1415926535897931 * (REAL(fVslider0::value()) / REAL(fVslider1))));
250 REAL fSlow1 = (1.0 / fSlow0);
251 REAL fSlow2 = (1.0 / (((fSlow1 + 1.0000000000000002) / fSlow0) + 1.0));
252 REAL fSlow3 = (1.0 / (fSlow1 + 1.0));
253 REAL fSlow4 = (1.0 - fSlow1);
254 REAL fSlow5 = (((fSlow1 + -1.0000000000000002) / fSlow0) + 1.0);
255 REAL fSlow6 = (2.0 * (1.0 - (1.0 / LowPass3_faustpower2_f(fSlow0))));
256 // Computed at runtime
257 for (int i = 0; (i < count); i = (i + 1)) {
258 REAL fTemp0 = REAL(input0[i]);
259 fVec0[0] = fTemp0;
260 fRec1[0] = (0.0 - (fSlow3 * ((fSlow4 * fRec1[1]) - (fTemp0 + fVec0[1]))));
261 fRec0[0] = (fRec1[0] - (fSlow2 * ((fSlow5 * fRec0[2]) + (fSlow6 * fRec0[1]))));
262 output0[i] = FAUSTFLOAT((fSlow2 * (fRec0[2] + (fRec0[0] + (2.0 * fRec0[1])))));
263 fVec0[1] = fVec0[0];
264 fRec1[1] = fRec1[0];
265 fRec0[2] = fRec0[1];
266 fRec0[1] = fRec0[0];
267 }

511// Down sample-rate adapter
512template <typename FILTER>
513class dsp_down_sampler : public sr_sampler<FILTER> {
514
515 public:
516
517 dsp_down_sampler(dsp* dsp):sr_sampler<FILTER>(dsp)
518 {}
519
520 virtual void init(int sample_rate)
521 {
522 this->fDSP->init(sample_rate / this->getFactor());
523 }
524
525 virtual void instanceInit(int sample_rate)
526 {
527 this->fDSP->instanceInit(sample_rate / this->getFactor());
528 }
529
530 virtual void instanceConstants(int sample_rate)
531 {
532 this->fDSP->instanceConstants(sample_rate / this->getFactor());
533 }
534
535 virtual dsp_down_sampler* clone() { return new dsp_down_sampler(decorator_dsp::clone()); }
536
537 virtual void compute(int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
538 {
539 int real_count = count / this->getFactor();
540
541 // Adapt inputs
542 FAUSTFLOAT** fInputs = (FAUSTFLOAT**)alloca(this->fDSP->getNumInputs() * sizeof(FAUSTFLOAT*));
543 for (int chan = 0; chan < this->fDSP->getNumInputs(); chan++) {
544 // Lowpass filtering in place on 'inputs'
545 this->fInputLowPass[chan].compute(count, inputs[chan], inputs[chan]);
546 // Allocate fInputs with 'real_count' frames
547 fInputs[chan] = (FAUSTFLOAT*)alloca(sizeof(FAUSTFLOAT) * real_count);
548 // Decimate
549 for (int frame = 0; frame < real_count; frame++) {
550 fInputs[chan][frame] = inputs[chan][frame * this->getFactor()];
551 }
552 }
553
554 // Allocate fOutputs with 'real_count' frames
555 FAUSTFLOAT** fOutputs = (FAUSTFLOAT**)alloca(this->fDSP->getNumOutputs() * sizeof(FAUSTFLOAT*));
556 for (int chan = 0; chan < this->fDSP->getNumOutputs(); chan++) {
557 fOutputs[chan] = (FAUSTFLOAT*)alloca(sizeof(FAUSTFLOAT) * real_count);
558 }
559
560 // Compute at lower rate
561 this->fDSP->compute(real_count, fInputs, fOutputs);
562
563 // Adapt outputs

607 virtual void compute(int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
608 {
609 int real_count = count * this->getFactor();
610
611 // Adapt inputs
612 FAUSTFLOAT** fInputs = (FAUSTFLOAT**)alloca(this->fDSP->getNumInputs() * sizeof(FAUSTFLOAT*));
613
614 for (int chan = 0; chan < this->fDSP->getNumInputs(); chan++) {
615 // Allocate fInputs with 'real_count' frames
616 fInputs[chan] = (FAUSTFLOAT*)alloca(sizeof(FAUSTFLOAT) * real_count);
617 // Puts zeros
618 memset(fInputs[chan], 0, sizeof(FAUSTFLOAT) * real_count);
619 for (int frame = 0; frame < count; frame++) {
620 // Copy one sample every 'UpFactor'
621 fInputs[chan][frame * this->getFactor()] = inputs[chan][frame];
622 }
623 // Lowpass filtering in place on 'fInputs'
624 this->fInputLowPass[chan].compute(real_count, fInputs[chan], fInputs[chan]);
625 }
626
627 // Allocate fOutputs with 'real_count' frames
628 FAUSTFLOAT** fOutputs = (FAUSTFLOAT**)alloca(this->fDSP->getNumOutputs() * sizeof(FAUSTFLOAT*));
629
630 for (int chan = 0; chan < this->fDSP->getNumOutputs(); chan++) {
631 fOutputs[chan] = (FAUSTFLOAT*)alloca(sizeof(FAUSTFLOAT) * real_count);
632 }
633
634 // Compute at upper rate
635 this->fDSP->compute(real_count, fInputs, fOutputs);
636
637 // Adapt outputs
638 for (int chan = 0; chan < this->fDSP->getNumOutputs(); chan++) {
639 // Lowpass filtering in place on 'fOutputs'
640 this->fOutputLowPass[chan].compute(real_count, fOutputs[chan], fOutputs[chan]);
641 // Decimate
642 for (int frame = 0; frame < count; frame++) {
643 // Apply volume
644 //outputs[chan][frame] = fOutputs[chan][frame * this->getFactor()] * this->getFactor();
645 outputs[chan][frame] = fOutputs[chan][frame * this->getFactor()];
646 }
647 }
648 }
