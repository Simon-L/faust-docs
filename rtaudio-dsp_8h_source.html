61 virtual int processAudio(double streamTime, void* inbuf, void* outbuf, unsigned long frames)
62 {
63 AVOIDDENORMALS;
64
65 float** inputs = (float**)alloca(fDSP->getNumInputs() * sizeof(float*));
66 float** outputs = (float**)alloca(fDSP->getNumOutputs() * sizeof(float*));
67
68 for (int i = 0; i < fDSP->getNumInputs(); i++) {
69 inputs[i] = &(static_cast<float*>(inbuf))[i * frames];
70 }
71 for (int i = 0; i < fDSP->getNumOutputs(); i++) {
72 outputs[i] = &(static_cast<float*>(outbuf))[i * frames];
73 }
74
75 // process samples
76 fDSP->compute(streamTime * 1000000., frames, inputs, outputs);
77 return 0;
78 }
79
80 static int audioCallback(void* outputBuffer, void* inputBuffer,
81 unsigned int nBufferFrames,
82 double streamTime, RtAudioStreamStatus status,
83 void* drv)
84 {
85 return static_cast<rtaudio*>(drv)->processAudio(streamTime, inputBuffer, outputBuffer, nBufferFrames);
86 }

122 bool init(const char* /*name*/, int numInputs, int numOutputs)
123 {
124 if (fAudioDAC.getDeviceCount() < 1) {
125 std::cout << "No audio devices found!\n";
126 return false;
127 }
128
129 RtAudio::DeviceInfo info_in = fAudioDAC.getDeviceInfo(fAudioDAC.getDefaultInputDevice());
130 RtAudio::DeviceInfo info_out = fAudioDAC.getDeviceInfo(fAudioDAC.getDefaultOutputDevice());
131 RtAudio::StreamParameters iParams, oParams;
132
133 iParams.deviceId = fAudioDAC.getDefaultInputDevice();
134 fDevNumInChans = info_in.inputChannels;
135 iParams.nChannels = fDevNumInChans;
136 iParams.firstChannel = 0;
137
138 oParams.deviceId = fAudioDAC.getDefaultOutputDevice();
139 fDevNumOutChans = info_out.outputChannels;
140 oParams.nChannels = fDevNumOutChans;
141 oParams.firstChannel = 0;
142
143 RtAudio::StreamOptions options;
144 options.flags |= RTAUDIO_NONINTERLEAVED;
145
146#if RTAUDIO_VERSION_MAJOR < 6
147 try {
148 fAudioDAC.openStream(((numOutputs > 0) ? &oParams : NULL),
149 ((numInputs > 0) ? &iParams : NULL), FORMAT,
150 fSampleRate, &fBufferSize, audioCallback, this, &options);
151 } catch (RtAudioError& e) {
152 std::cout << '\n' << e.getMessage() << '\n' << std::endl;
153 return false;
154 }
155 return true;
156#else
157 RtAudioErrorType err = fAudioDAC.openStream(
158 ((numOutputs > 0) ? &oParams : NULL),
159 ((numInputs > 0) ? &iParams : NULL), FORMAT,
160 fSampleRate, &fBufferSize, audioCallback, this, &options);
161 if (err != RTAUDIO_NO_ERROR) {
162 std::cout << '\n' << fAudioDAC.getErrorText() << '\n' << std::endl;
163 return false;
164 }
165 return true;
166#endif
167 }

169 void setDsp(::dsp* DSP)
170 {
171 fDSP = DSP;
172
173 if (fDSP->getNumInputs() > fDevNumInChans || fDSP->getNumOutputs() > fDevNumOutChans) {
174 printf("DSP has %d inputs and %d outputs, physical inputs = %d physical outputs = %d \n",
175 fDSP->getNumInputs(), fDSP->getNumOutputs(),
176 fDevNumInChans, fDevNumOutChans);
177 fDSP = new dsp_adapter(fDSP, fDevNumInChans, fDevNumOutChans, fBufferSize);
178 }
179
180 fDSP->init(fSampleRate);
181 }
