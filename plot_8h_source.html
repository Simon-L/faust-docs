406 void addPoint(double x, double y, bool alwaysInclude=false) {
407 if (std::isnan(x) || std::isnan(y)) return;
408 auto clip = clipStack.back();
410 char mask = (clip.left > x)
411 | (2*(clip.right < x))
412 | (4*(clip.top > y))
413 | (8*(clip.bottom < y));
414 if (alwaysInclude) mask = 0;
415 outOfBoundsMask &= mask;
416 if (!outOfBoundsMask) {
417 if (pointState == PointState::outOfBounds) {
418 // Draw the most recent out-of-bounds point
419 raw(" ", round(prevPoint.x), " ", round(prevPoint.y));
420 lastDrawn = prevPoint;
421 pointState = PointState::singlePoint;
422 }
423 if (pointState == PointState::singlePoint) {
424 pointState = PointState::pendingLine;
425 totalPendingError = 0;
426 } else if (pointState == PointState::pendingLine) {
427 // Approximate the pending point as being on the line from last-drawn point -> current
428 double d1 = std::hypot(prevPoint.x - lastDrawn.x, prevPoint.y - lastDrawn.y);
429 double d2 = std::hypot(x - lastDrawn.x, y - lastDrawn.y);
430 double scale = d2 ? d1/d2 : 0;
431 double extX = lastDrawn.x + (x - lastDrawn.x)*scale;
432 double extY = lastDrawn.y + (y - lastDrawn.y)*scale;
433 // How far off would that be?
434 totalPendingError += std::hypot(extX - prevPoint.x, extY - prevPoint.y);
435 if (totalPendingError > invPrecision) {
436 // Would be too much accumulated error. Draw the pending segment, and start a new one.
437 raw(" ", round(prevPoint.x), " ", round(prevPoint.y));
438 lastDrawn = prevPoint;
439 totalPendingError = 0;
440 }
441 } else { // start
442 raw(" ", round(x), " ", round(y));
443 lastDrawn = {x, y};
444 pointState = PointState::singlePoint;
445 }
446 outOfBoundsMask = mask;
447 if (outOfBoundsMask) {
448 if (pointState == PointState::pendingLine) {
449 raw(" ", round(prevPoint.x), " ", round(prevPoint.y));
450 }
451 raw(" ", round(x), " ", round(y)); // Draw the first out-of-bounds point
452 pointState = PointState::outOfBounds;
453 }
454 }
455 prevPoint = {x, y};

1098 this->drawAt = {sx, sy};
1099 } else {

1101 double ax = std::cos(angle), ay = std::sin(angle);
1102
1103 double px = sx + distance*ax, py = sy + distance*ay;
1104 double tx = px, ty = py;
1105 double fontSize = style.labelSize;
1106 double letterHeight = fontSize*0.8;

1115 this->alignment = 1;
1116 tx += space;
1117 ty += ay*verticalWiggle;
1118 } else if (ay > 0) {
1119 ty += letterHeight;
1120 tx += ax*fontSize;
1121 this->alignment = ax;
1122 } else {
1123 ty -= letterHeight;
1124 tx += ax*fontSize;
1125 this->alignment = ax;
1126 }
1127
1128 double lineDistance = distance - space;
1129 drawLineFrom = drawLineTo = {px, py};
1130 if (lineDistance > space) {
1131 drawLineTo = {sx + ax*space, sy + ay*space};
1132 }
