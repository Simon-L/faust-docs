63 void init(dsp* mono_dsp, audio* driver, midi_handler* handler)
64 {
65 bool midi_sync = false;
66 bool midi = false;
67 int nvoices = 0;
68 fRunning = false;
69
70 MidiMeta::analyse(mono_dsp, midi, midi_sync, nvoices);
71
72 // Getting the UI JSON
73 JSONUI jsonui1(mono_dsp->getNumInputs(), mono_dsp->getNumOutputs());
74 mono_dsp->buildUserInterface(&jsonui1);
75 fJSONUI = jsonui1.JSON();
76
77 // Getting the metadata JSON
78 JSONUI jsonui1M(mono_dsp->getNumInputs(), mono_dsp->getNumOutputs());
79 mono_dsp->metadata(&jsonui1M);
80 fJSONMeta = jsonui1M.JSON();
81
82 if (nvoices > 0) {
83
84 fPolyDSP = new mydsp_poly(mono_dsp, nvoices, true);
85
86 #if POLY2
87 fFinalDSP = new dsp_sequencer(fPolyDSP, new effect());
88 #else
89 fFinalDSP = fPolyDSP;
90 #endif
91
92 // Update JSONs with Poly version
93 JSONUI jsonui2(mono_dsp->getNumInputs(), mono_dsp->getNumOutputs());
94 fFinalDSP->buildUserInterface(&jsonui2);
95 fJSONUI = jsonui2.JSON();
96
97 JSONUI jsonui2M(mono_dsp->getNumInputs(), mono_dsp->getNumOutputs());
98 fFinalDSP->metadata(&jsonui2M);
99 fJSONMeta = jsonui2M.JSON();
100
101 } else {
102 fPolyDSP = nullptr;
103 fFinalDSP = mono_dsp;
104 }
105
106 fFinalDSP->buildUserInterface(&fMidiUI);
107 fFinalDSP->buildUserInterface(&fAPIUI);
108
109 // Retrieving DSP object name
110 struct MyMeta : public Meta
111 {
112 std::string fName;
113 void declare(const char* key, const char* value)
114 {
115 if (strcmp(key, "name") == 0) fName = value;
116 }
117 MyMeta():fName("Dummy"){}
118 };
119
120 MyMeta meta;
121 fFinalDSP->metadata(&meta);
122 if (handler) handler->setName(meta.fName);
123
124 if (driver) {
125 // If driver cannot be initialized, start will fail later on...
126 if (!driver->init(meta.fName.c_str(), fFinalDSP)) {
127 delete driver;
128 fDriver = nullptr;
129 } else {
130 fDriver = driver;
131 }
132 } else {
133 fDriver = nullptr;
134 }
135 }

143 }

186 /*
187 * keyOn(pitch, velocity)
188 * Instantiates a new polyphonic voice where velocity
189 * and pitch are MIDI numbers (0-127). keyOn can only
190 * be used if nvoices > 0. keyOn will return 0 if the
191 * object is not polyphonic and the allocated voice otherwise.
192 */
193 MapUI* keyOn(int pitch, int velocity)
194 {
195 if (fPolyDSP) {
196 return fPolyDSP->keyOn(0, pitch, velocity); // MapUI* passed to Java as an integer
197 } else {
198 return 0;
199 }
200 }
201
202 /*
203 * keyOff(pitch)
204 * De-instantiates a polyphonic voice where pitch is the
205 * MIDI number of the note (0-127). keyOff can only be
206 * used if nvoices > 0. keyOff will return 0 if the
207 * object is not polyphonic and 1 otherwise.
208 */
209 int keyOff(int pitch, int velocity = 0)
210 {
211 if (fPolyDSP) {
212 fPolyDSP->keyOff(0, pitch, velocity);
213 return 1;
214 } else {
215 return 0;
216 }
217 }

266 /*
267 * Propagate MIDI data to the Faust object.
268 */
269 void propagateMidi(int count, double time, int type, int channel, int data1, int data2)
270 {
271 if (count == 3) fMidiHandler.handleData2(time, type, channel, data1, data2);
272 else if (count == 2) fMidiHandler.handleData1(time, type, channel, data1);
273 else if (count == 1) fMidiHandler.handleSync(time, type);
274 // In POLY mode, update all voices
275 GUI::updateAllGuis();
276 }
