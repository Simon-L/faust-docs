56class esp32_midi : public midi_handler {
57
58 private:

60 TaskHandle_t fProcessMidiHandle;

62 void processMidi()
63 {
64 double time = 0;
65 uint8_t data[RX_BUF_SIZE];
66 MIDIParser parser;
67 jdksMIDIMessage message;
68
69 while (true) {
70 int rxBytes = uart_read_bytes(UART_NUM_1, data, RX_BUF_SIZE, 1);
71 if (rxBytes > 0) {
72 for (int i = 0; i < rxBytes; i++) {
73 if (parser.Parse((uchar)data[i], &message)) {
74 unsigned char status = message.GetStatus();
75 if (status < 0xF0)
76 { // channel/system message discriminator.
77 unsigned char type = message.GetType();
78 switch (type)
79 {
80 case 0x80: // Note Off
81 handleKeyOff(time, message.GetChannel(), message.GetNote(), message.GetVelocity());
82 break;
83 case 0x90: // Note On
84 handleKeyOn(time, message.GetChannel(), message.GetNote(), message.GetVelocity());
85 break;
86 case 0xA0: // Poly Key Pressure
87 handlePolyAfterTouch(time, message.GetChannel(), message.GetNote(), message.GetByte2());
88 break;
89 case 0xB0: // Control Change
90 handleCtrlChange(time, message.GetChannel(), message.GetController(), message.GetControllerValue());
91 break;
92 case 0xC0: // Program Change // No Bank Select in faust?
93 handleProgChange(time, message.GetChannel(), message.GetPGValue());
94 break;
95 case 0xD0: // Channel Pressure
96 handleAfterTouch(time, message.GetChannel(), message.GetChannelPressure());
97 break;
98 case 0xE0: // Pitch Bend
99 handlePitchWheel(time, message.GetChannel(), message.GetByte1(), message.GetByte2());
100 break;
101 default:
102 break;
103 }
104 }
105 else
106 {
107 switch (status)
108 {
109 case 0xF8: // Timing Clock
110 handleClock(time);
111 break;
112 // We can consider start and continue as identical messages.
113 case 0xFA: // Start
114 case 0xFB: // Continue
115 handleStart(time);
116 break;
117 case 0xFC: // Stop
118 handleStop(time);
119 break;
120 case 0xF0: // SysEx Start
121 // TODO
122 break;
123 case 0xF7: // SysEx Stop
124 // TODO
125 break;
126 default:
127 break;
128 }
129 }
130 }
131 // Synchronize all GUI controllers
132 GUI::updateAllGuis();
133 }
134 }
135 }
136 }

138 static void processMidiHandler(void* arg)
139 {
140 static_cast<esp32_midi*>(arg)->processMidi();
141 }
142
143 public:
144
145 esp32_midi():midi_handler("esp32"),fProcessMidiHandle(NULL)
146 {
147 // Setup UART for MIDI
148 const uart_config_t uart_config = {
149 .baud_rate = 31250,
150 .data_bits = UART_DATA_8_BITS,
151 .parity = UART_PARITY_DISABLE,
152 .stop_bits = UART_STOP_BITS_1,
153 .flow_ctrl = UART_HW_FLOWCTRL_DISABLE,
154 .rx_flow_ctrl_thresh = 122,
155 .use_ref_tick = false
156 };
157 uart_param_config(PORT_NUM, &uart_config);
158 uart_set_pin(PORT_NUM, TX1, RX1, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
159 // We won't use a buffer for sending data.
160 uart_driver_install(PORT_NUM, RX_BUF_SIZE * 2, 0, 0, NULL, 0);
161 }
162
163 virtual ~esp32_midi()
164 {
165 stopMidi();
166 }
167
168 bool startMidi()
169 {
170 // Start MIDI receive task
171 return (xTaskCreatePinnedToCore(processMidiHandler, "Faust MIDI Task", 4096, (void*)this, 5, &fProcessMidiHandle, 1) == pdPASS);
172 }
173
174 void stopMidi()
175 {
176 if (fProcessMidiHandle != nullptr) {
177 vTaskDelete(fProcessMidiHandle);
178 fProcessMidiHandle = nullptr;
179 }
180 }
