35double LookupTable::getValue(double x)
36{
37 // Note: Assumes points are monotonically increasing in X!
38
39 int i=0;
40 while (x>m_Points[i*2] && i<m_nPoints)
41 i++;
42
43 if (i==0)
44 return m_Points[1];
45
46 if (i>=m_nPoints)
47 return m_Points[(m_nPoints-1)*2+1];
48
49 double ratio =
50 (x - m_Points[(i-1)*2])
51 / (m_Points[i*2] - m_Points[(i-1)*2]);
52
53 return m_Points[(i-1)*2+1]*(1-ratio) + m_Points[i*2+1]*(ratio);
54}
