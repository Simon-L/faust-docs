46struct aliastarget
47{
48 double fMinIn;
49 double fMaxIn;
50 double fMinOut;
51 double fMaxOut;
52 std::string fTarget; // the real osc address
53
54 aliastarget(const char* address, double imin, double imax, double omin, double omax)
55 : fMinIn(imin), fMaxIn(imax), fMinOut(omin), fMaxOut(omax), fTarget(address) {}
56
57 aliastarget(const aliastarget& t)
58 : fMinIn(t.fMinIn), fMaxIn(t.fMaxIn), fMinOut(t.fMinOut), fMaxOut(t.fMaxOut), fTarget(t.fTarget) {}
59
60 // explicit copy assignment operator
61 aliastarget& operator=(const aliastarget& other) { return *this; }
62
63 double scale(double x) const
64 {
65 if (fMinIn < fMaxIn) {
66 // increasing control
67 double z = (x < fMinIn) ? fMinIn : (x > fMaxIn) ? fMaxIn : x;
68 return fMinOut + (z-fMinIn)*(fMaxOut-fMinOut)/(fMaxIn-fMinIn);
69
70 } else if (fMinIn > fMaxIn) {
71 // reversed control
72 double z = (x < fMaxIn) ? fMaxIn : (x > fMinIn) ? fMinIn : x;
73 return fMinOut + (fMinIn-z)*(fMaxOut-fMinOut)/(fMinIn-fMaxIn);
74
75 } else {
76 // no control !
77 return (fMinOut+fMaxOut)/2.0f;
78 }
79 }
80
81 double invscale(double x) const
82 {
83 if (fMinOut < fMaxOut) {
84 // increasing control
85 double z = (x < fMinOut) ? fMinOut : (x > fMaxOut) ? fMaxOut : x;
86 return fMinIn + (z-fMinOut)*(fMaxIn-fMinIn)/(fMaxOut-fMinOut);
87
88 } else if (fMinOut > fMaxOut) {
89 // reversed control
90 double z = (x < fMaxOut) ? fMaxOut : (x > fMinOut) ? fMinOut : x;
91 return fMinIn + (fMinOut-z)*(fMaxIn-fMinIn)/(fMinOut-fMaxOut);
92
93 } else {
94 // no control !
95 return (fMinIn+fMaxIn)/2.0f;
96 }
97 }
98};
